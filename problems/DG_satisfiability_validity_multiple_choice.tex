\begin{problem}%\label{generprob}

\begin{problemparts}
For the following predicates, indicate whether they are valid (V),
satisfiable but not valid (S), or unsatisfiable (N). For valid and/or
satisfiable predicates, give an assignment of truth values that would
satisfy the predicate.

\problempart P implies P
\begin{solution}
(V)
\end{solution}

\problempart P implies notP
\begin{solution}
(S)
\end{solution}

\problempart P and Q or (P implies Q)
\begin{solution}
(S)
\end{solution}

\problempart P and Q and Z and (P implies notQ)
\begin{solution}
(N)
\end{solution}

\problempart P and Q and (notP or notQ) implies (Z and (P implies Q) or (P and notP))
\begin{solution}
(V)
\end{solution}

\problempart notP or (P implies Q)
\begin{solution}
(V)
\end{solution}

\problempart P or (notP or Q) implies (Q and Q implies notQ)
\begin{solution}
(N)
\end{solution}

\end{problemparts}

\end{problem}
