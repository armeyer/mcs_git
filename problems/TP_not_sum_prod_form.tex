\begin{problem}
%\bparts

%\ppart
Give an example of a propositional formula involving no
operations other than $\QAND, \QOR, \QNOT$ and neither of the
truth-constants \True\ or \False, which is in neither sum-of-products
(disjunctive) form nor product-of-sums (conjunctive) form.

\begin{solution}
\[
\QNOT(P \QOR Q)
\]
or any other formula where a connective appears in the scope of a negation.
Also,
\[
(P \QOR (Q \QAND (R \QOR S)))
\]
or any other formula where $\QAND$'s and $\QOR$'s alternate to depth more
than two.

\end{solution}
\iffalse
\ppart Which of the following are examples of formulas that are in both
sum-of-products and product-of-sums form.
\begin{enumerate}
\item $P$\label{MQ916_P}
\item $P \QOR \bar{Q}$ \label{MQ916_PORQ}
\item $\bar{P} \QAND \bar{Q}$ \label{MQ916_PANDQ}
\item $(P \QOR \bar{Q}) \QAND (\bar{P} \QOR \bar{Q})$
\item $(P \QAND \bar{Q}) \QOR \bar{P} \QOR \bar{Q}$
\end{enumerate}

\begin{solution}
\ref{MQ916_P}, \ref{MQ916_PORQ}, \ref{MQ916_PANDQ}
\end{solution}
\eparts
\fi
