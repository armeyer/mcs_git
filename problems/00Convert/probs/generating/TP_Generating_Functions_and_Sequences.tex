\begin{problem} \textbf{Generating Functions and Sequences}

Write a formula for the generating function of each of the following
sequences.

\bparts

\ppart
0, 0, 1, 1, 1,\dots

\begin{solution}
$x^2/(1-x)$
\end{solution}

\ppart
1, 1, 0, 0, 0,\dots

\begin{solution}
$1+x$
\end{solution}

\ppart
1, 0, 1, 0, 1, 0, 1,\dots

\begin{solution}
$1/(1-x^2)$
\end{solution}

\ppart
1, 4, 6, 4, 1, 0, 0, 0,\dots

\begin{solution}
$(1+x)^4$
\end{solution}

\ppart
1, 1, 1/2, 1/6, 1/24, 1/120,\dots

\begin{solution}
$e^{x}$
\end{solution}

\ppart
1, 2, 3, 4, 5,\dots

\begin{solution}
$1/(1-x)^2$, the derivative of $1/(1-x)$.
\end{solution}

\ppart
1, 4, 9, 16, 25,\dots

\begin{solution}
$(1+x)/(1-x)^3$, the derivative of $x/(1-x)^2$.
\end{solution}

\eparts

\end{problem}
