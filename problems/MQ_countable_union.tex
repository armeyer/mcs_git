\begin{problem}
Prove that if $A_0, A_1,\dots, A_n, \dots $ is an infinite
sequence of countable sets, then so is
\[
\lgunion_{n=0}^\infty A_n
\]

\begin{staffnotes}
If needed, there is an almost give-away

\hint Use the fact that $\nngint \cross \nngint$ is countable.
\end{staffnotes}

\begin{solution}
Since $A_m$ is countable, there is a list of its elements:
\[
a_{(m,0)}, a_{(m,1)}, \dots, a_{(m,n)}, \dots
\]
Let $f:\nngint \to \nngint^2$ be a bijection, then
\[
a_{f(0)}, a_{f(0)}, \dots, a_{f(n)}, \dots
\]
is a list of $\lgunion_{n=0}^\infty A_n$.

\end{solution}
\end{problem}
