\begin{problem}

\bparts \mbox{}

\inbook{\ppart Write a CNF (Conjunctive Normal Form) formula for
\[
P \QIFF Q.
\]

%\examspace[0.5in]

\begin{solution}
\[
(p \QOR \bar{Q}) \QAND (\bar{Q} \QOR P)
\]
\end{solution}
}

\ppart Bernardo, Francisco, Marcellus and Horatio are standing guard on the tower wall when they hear a ghost. Someone needs to leave to investigate the ghost, but someone needs to keep standing guard on the tower wall. Let $B$ be the proposition that  \textbf{B}ernardo will leave to investigate the ghost, $F$ that \textbf{F}rancisco will leave to investigate, $M$ that  \textbf{M}arcellus will leave to investigate, and $H$ that  \textbf{H}oratio will leave to investigate. 

 Using these propositions, translate the following sentence
directly into a Full\footnote{``Full'' means each $\QOR$ term
  contain all of $B, F, M$, and $H$.} Conjunctive Normal Form
propositional formula
\begin{equation}\tag{*}
\text{At least one guard will leave to investigate, and at least one guard will not leave.}
\end{equation}
\examspace[1in]

\begin{solution}
\[
(\QOR\ B\ F\ M\ H\ ) \QAND (\QOR \bar{B}\ \bar{F}\ \bar{M}\ \bar{H}\)
\]
\end{solution}

\examspace[1in]

\ppart How many \QAND-of-literals subformulas are there in the Full\footnote{``Full'' means each $\QAND$ term
  contain all of $B, F, M$, and $H$.} 
\textbf{Dis}junctive Normal Form that expresses~(*)?

\begin{center}
\exambox{0.6in}{0.6in}{-0.1in}
\end{center}


\begin{solution}
$14 = 2^4 -2$.

This is the number of \True\ entries in the 16 row truth table
for~(*), since there are only two \False\ entries.
\end{solution}

\eparts

\end{problem}
