\begin{problem}
Let
\[
f_0, f_1, f_2, \dots, f_k,\dots
\]
be an infinite sequence of (not necessarily different) nonnegative
real-valued total functions $f_k:\nngint \to \reals^{\geq 0}$.  Define
the function $s: \nngint \to \reals^{\geq 0}$ by the rule:
\[
s(n) \eqdef \sum_{k \leq \sqrt{n}} f_k(n).
\]

\bparts

\ppart There is a function $b(k)$ such that from $b(k)$ on, $s$ is at
least as big as $f_k$:
\[
\forall k \forall n \geq b(k).\ s(n) \geq f_k(n).
\]
Which functions of $k$ below are examples of a possible $b(k)$?
\begin{enumerate}[i]
\item $k$
\item $\sqrt{k}$
\item $\sqrt{n}$
\item $n^2$
\item $k^2$
\item $k^3$
\item $2^k$
\end{enumerate}

\begin{solution}
By definition, $s(n) \geq f_k(n)$ for $n \geq \sqrt{k}$, so
\[
b(k) \eqdef \sqrt{k},
\]
will work, as will any other function $\geq \sqrt{k}$.  So all
the above choices work except for the third and fourth ($\sqrt{n},
n^2$) which are nonsense.
\end{solution}

\ppart Which properties of the sequence of $f_k$'s ensures that $s$
does not appear in the sequence?

\begin{enumerate}[i]
\item $f_7$ is positive ($\forall n.\, f_7(n) > 0$).

\item $f_7$ is positive infinitely often ($\forall k \exists n.\, f_7(n) > 0$).

\item $f_7$ has no upper bound.

\item $f_7(n) = n^2$.

\item $f_7$ and $f_8$ are positive.

\item $f_7$ and $f_8$ are positive infinitely often (not necessarily
  at the same places),

\item neither $f_7$ nor $f_8$ has an upper bound.

\end{enumerate}

\begin{solution}
None of the first four choices prevent $f_{k\neq
  7}(n) = 0$ for all $n$, which case $s = f_7$ is in the sequence.

The remaining choices ensure that for every $k$, 
\[
s(n) \geq 1+ f_k(n)
\] 
for infinitely many $n$, so $s \neq f_k$ for any $k$.
\end{solution}

\eparts

\end{problem}
