\begin{problem}
For each of the following relations, decide whether it is reflexive,
whether it is symmetric, whether it is transitive, and whether it is
an equivalence relation.

\bparts

\ppart
$\set{ (a,b) \suchthat a \text{ and } b \text{ are the same age} }$

\begin{solution}
Reflexive, symmetric, transitive, equivalence.

\end{solution}

\ppart
$\set{ (a,b) \suchthat a\text{ and }b\text{ have the same parents}}$

\begin{solution}

Reflexive, symmetric, transitive, equivalence.

\end{solution}

\ppart
$\set{ (a,b) \suchthat a\text{ and }b\text{ speak a common language}}$

\begin{solution}
Reflexive, symmetric.

\end{solution}

\eparts

\end{problem}
