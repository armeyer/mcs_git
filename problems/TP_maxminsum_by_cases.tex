\begin{problem}
Prove by cases that
\begin{equation}\tag{*}%\label{minrsplus}
\max(r,s) + \min(r,s) = r + s
\end{equation}
for all real numbers $r,s$.

\begin{solution}
\inductioncase{Case 1}: ($r \geq s$).  In this case, $\max(r,s) = r$
and $\min(r,s) = s$, so (*) becomes %~\eqref{minrsplus}
\[
r + s = r + s,
\]
which is certainly true.

\inductioncase{Case 2}: ($s \geq r$).  In this case, $\max(r,s) = s$ and
$\min(r,s) = r$, so (*) becomes %~\eqref{minrsplus}
\[
s+r = r + s,
\]
which is certainly true.

So in any case,~(*) is true.  %\eqref{minrsplus}
\end{solution}
\end{problem}
