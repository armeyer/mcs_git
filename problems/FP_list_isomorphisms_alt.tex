\begin{problem}
  \bparts

List \emph{all} of the isomorphisms between the two graphs in
Figure~\ref{iso-examples}. Explain why there are no others.

\begin{figure}[h]
\graphic{iso-examples}
\caption{Graphs with several isomorphisms}
\label{iso-examples}
\end{figure}

\begin{solution}
  The four isomorphisms have the following vertex correspondences:
  \begin{align*}
    bq, ap, ct, dr, es \\
    bq, ap, ct, ds, er \\
    bq, at, cp, dr, es \\
    bq, at, cp, ds, er.
  \end{align*}

  Recall that nodes related by an isomorphism must have equal degree, so in any isomorphism between $G$ and $H$, vertex $b$ must map to the only degree-$2$ vertex in $H$, namely, $q$. Then, $b$'s neighbors, $a$ and $c$, must map to $q$'s neighbors, $p$ and $t$, in either order. Finally, the remaining two vertices, $d$ and $e$, must map to the remaining vertices, $r$ and $s$. This leaves the four possibilities listed above, and all four of these may be checked to be valid isomorphisms.
  \end{solution}

\end{problem}
