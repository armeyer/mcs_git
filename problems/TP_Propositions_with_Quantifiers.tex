\begin{problem}
For each of the following propositions:
\begin{enumerate}[(i)]

\item $\forall x\,  \exists y. \; 2x-y = 0$,

\item $\forall x\, \exists y. \; x-2y = 0$,

\item
$\forall x.\, x < 10 \QIMPLIES (\forall y. \; y < x \QIMPLIES y < 9)$,

\item
$\forall x\, \exists y.\; [y > x \And \exists z. \; y+z = 100]$,

\end{enumerate}
indicate which propositions are \False\ when the variables range over:

\bparts

\ppart
the nonnegative integers.  \hfill\examrule[2in]

\begin{staffnotes}
The explanations given are minimal.  Your team should try to explain
(briefly) the truth falsehood of all the assertions.
\end{staffnotes}

\begin{solution}
Counterexample for 2:  let $x = 1$.

Counterexample for 4:  let $x = 100$.
\end{solution}

\ppart
the integers. \hfill\examrule[2in]

\begin{solution}
Counterexample for 2:  let $x = 1$.
\end{solution}

\ppart
the real numbers. \hfill\examrule[2in]

\begin{solution}
Counterexample for 3:  let $x = 9.5$.
\end{solution}

\eparts

\end{problem}
