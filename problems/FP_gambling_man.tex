\begin{problem}
  \Albert\ has a gambling problem.  He plays 240 hands of draw poker, 120 hands
  of black jack, and 40 hands of stud poker per day.  He wins a hand of
  draw poker with probability 1/6, a hand of black jack with probability
  1/2, and a hand of stud poker with probability 1/5.  Let $W$ be the
  expected number of hands that \Albert\ wins in a day.

\bparts

\ppart What is $\expect{W}$?

\exambox{1.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
The expectation is the sum of the expectations of all the individual hands, namely
\[
240(1/6)+120(1/2)+40(1/5)=108.
\]
\end{solution}

\ppart What would the Markov bound be on the probability that
\Albert\ will win at least 216 hands on a given day?

\exambox{0.5in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
The expected number of games won is 108, so
by Markov, $\pr{W \geq 216} \leq 108/216 = 1/2$.
\end{solution}

\ppart Assume the outcomes of the card games are pairwise independent.
What is $\variance{W}$?  You may answer with a numerical expression
that is not completely evaluated.

\exambox{3.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
The variance can also be calculated using linearity of variance.
For an individual hand the variance is $p(1-p)$ where $p$ is the
probability of winning.  Therefore the variance is
\[
240(1/6)(5/6) + 120(1/2)(1/2) + 40(1/5)(4/5) = 1046/15 = 69\ \tfrac{11}{15}.
\]
\end{solution}

\ppart What would the Chebyshev bound be on the probability that
\Albert\ will win at least 216 hands on a given day?  You may answer
with a numerical expression that includes the constant $v =
\variance{W}$.

\exambox{2.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
\begin{align*}
\pr{W \geq 216} & = \pr{W - 108 \geq 108} \leq \pr{\abs{W - 108} \geq 108}\\
                & \leq \frac{v}{108^2} = \frac{1046}{15(108)^2} \approx 0.005970.
\end{align*}

(A very slightly better bound of 0.005943 comes from using the
one-sided Chebyshev bound from Problem~\bref{PS_Chebyshev_one_sided}.)

\end{solution}
\eparts
\end{problem}
