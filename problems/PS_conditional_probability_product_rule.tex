\begin{problem}
The Conditional Probability Product Rule for $n$ Events is
\begin{rul*}[]
\begin{align*}
\pr{E_1 \intersect E_2 \intersect \dots \intersect E_n}
   =& \pr{E_1}
        \cdot \prcond{E_2}{E_1}
        \cdot \prcond{E_3}{E_1 \intersect E_2}
        \cdots \\
    &\quad\cdot
        \prcond{E_n}{E_1 \intersect E_2 \intersect \dots
          \intersect E_{n - 1}}\, .
\end{align*}
\end{rul*}

\bparts

\ppart Restate the Rule without elipses---that is, without
``\dots''---by using ``$\lgintersect_{i=a}^b$'' for suitable $a,b$.

\begin{solution}
$\pr{\lgintersect_{i=1}^n E_i}
   = \prod_{i=1}^n \prcond{E_i}{\lgintersect_{j=1}^{i-1} E_j}$

The convention that an empty intersection is the
whole space, in particular,
\[
\lgintersect_{j=1}^{0} E_j = \sspace,
\]
ensures that this equation even hold for $n=1$.\footnote{It even holds
  for $n=0$ using the convention that an empty product equals 1.}
\end{solution}

\ppart Prove it by induction.

\end{problem}
