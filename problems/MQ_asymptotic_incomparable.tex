\begin{problem} Let $f,g: \nngint \to \nngint$ be defined as
\[
f(n) \eqdef  n^n, \qquad g(n) \eqdef \begin{cases}
                                     n^{n-(1/2)} & \text{if $n$ is odd},\\
                                     n^{n+(1/2)} & \text{if $n$ is even}.
                                    \end{cases}
\]

\begin{staffnotes}
Note that both $f$ and $g$ are strictly increasing.
\end{staffnotes}

\bparts

\ppart  Verify that $f \neq O(g)$.

\examspace[2.0in]

\begin{solution}
\[
\limsup_{n \to \infty} \frac{f}{g} = \lim \frac{n^n}{n^{n-(1/2)}}= \lim
\frac{n^n}{(1/\sqrt{n})n^n}= \sqrt{n} = \infty,
\]
so $f \neq O(g)$.

\end{solution}

\ppart Verify that $g \neq O(f)$.

\examspace[2.0in]

\begin{solution}
\[
\limsup_{n \to \infty} \frac{g}{f} = \lim \frac{n^{n+(1/2)}}{n^n}= \lim
\frac{\sqrt{n}\, n^n}{n^n}= \sqrt{n} = \infty,
\]
so $g \neq O(f)$.

\end{solution}

\eparts

\end{problem}
