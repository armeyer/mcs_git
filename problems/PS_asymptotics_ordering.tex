\begin{problem}

Arrange the following functions in a sequence $f_1$, $f_2$, ... $f_{24}$ so that $f_i = O(f_{i+1})$. Additionally, if $f_i = \Theta(f_{i+1})$, indicate that too:

\begin{enumerate}
\item $n\log{n}$
\item $2^{100}n$
\item $n^{-1}$
\item $n^{-1/2}$
\item $(\log{n})/n$
\item $n \choose 64$
\item $n!$
\item $2^{2^{100}}$
\item $2^{2^n}$
\item $2^n$
\item $3^n$
\item $n2^n$
\item $2^{n+1}$
\item $2n$
\item $3n$
\item $\log{(n!)}$
\item $\log_2{n}$
\item $\log_{10}{n}$
\item $2.1^{\sqrt{n}}$
\item $2^{2n}$
\item $4^n$
\item $n^{64}$
\item $n^{65}$
\item $n^n$

\end{enumerate}

\begin{solution}

The correct ordering is the following:
\begin{enumerate}
\item $n^{-1}$
\item $(\log{n})/n$
\item $n^{-1/2}$
\item $2^{2^{100}}$
\item $\log_{10}{n} = \Theta(\log_2{n})$
\item $2^{100}n = \Theta(2n) = \Theta(3n)$
\item $\log{(n!)} = \Theta(n\log{n})$
\item ${n \choose 64} = \Theta(n^{64})$
\item $n^{65}$
\item $2.1^{\sqrt{n}}$
\item $2^n = \Theta(2^{n+1})$
\item $n2^n$
\item $3^n$
\item $4^n = \Theta(2^{2n})$
\item $n!$
\item $n^n$
\item $2^{2^n}$
\end{enumerate}

\end{solution}


\end{problem}
