\begin{problem}
Use the Euclidean Algorithm to prove that
\[
\gcd(13a+8b, 5a+3b) = \gcd(a, b).
\]

\begin{solution}
\begin{align*}
\gcd(13a+8b, 5a+3b)
   & = \gcd(5a+3b,3a+2b)\\
      &\qquad (\text{since } 13a+8b = 2\cdot (5a+3b) + (3a+2b)\ )\\ 
   & = \gcd(3a+2b, 2a+b)\\
      &\qquad (\text{since } 5a+3b = 1\cdot (3a+2b) + (2a+b)\ )\\
   & = \gcd(2a+b, a+b)\\
      &\qquad (\text{since } 3a+2b = 1\cdot (2a+b) + (a+b)\ )\\
   & = \gcd(a,a+b)\\
      &\qquad (\text{since } 2a+b = 1\cdot (a+b) + a\ )\\
   & = \gcd(a,b)\\
      &\qquad (\text{since } a+b =   1\cdot a+ b\ )
\end{align*}

\end{solution}

\end{problem}
