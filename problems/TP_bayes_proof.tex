\begin{problem}
State and prove a version of Bayes Rule that applies to disjoint
events $E_1, \dots, E_n$ whose union is the whole sample space.
You may assume the $n$-event Law of Total Probability,
\inbook{Problem~\bref{TP_multi_total_prob}.}%
\inhandout{\[
\pr{A} = \sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}.
\]
}

\begin{solution}
\begin{rul*}[Bayes' Rule, $n$-events]
\[ \prcond{E_1}{A} = \frac{\prcond{A}{E_1}\cdot
    \pr{E_1}}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
\]
\end{rul*}

\begin{proof}
\begin{align*}
\prcond{E_1}{A}
  & \eqdef \frac{E_1 \intersect A_1}{\pr{A}}\\
  & =  \frac{E_1 \intersect A_1}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
       & \text{(Total probability)}\\
  & =  \frac{\prcond{A}{E_1}\cdot \pr{E_1}}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
       & \text{\text{(def conditional prob)}}
\end{align*}

\end{proof}

\end{solution}
\end{problem}
