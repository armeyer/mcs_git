\begin{problem}
\iffalse
Suppose that $R$ is a random variable.  Let $C$ be an event.  he
\emph{conditional expectation $\expcond{R}{C}$ of $R$ given
event $C$} is defined by:
\[
\expcond{R}{C} \eqdef \sum_{x \in \range{R}} x \cdot \prcond{R = x}{C}.
\]
\fi

\bparts

\ppart Compute the expected value of the number rolled on a
fair, six-sided die, given that the outcome is even.

\examspace[1.0in]
\exambox{0.5in}{0.5in}{0in}

\begin{solution}
Let the random variable $R$ be the number rolled, and let
$C$ be the event that the number rolled is even.
\begin{align*}
\expcond{R}{C}  & =    \sum_{x=1}^{6} x \cdot \prcond{R = x}{C} \\
                & =    1 \cdot 0 + 2 \cdot \frac{1}{3} + 
                       3 \cdot 0 + 4 \cdot \frac{1}{3} + 
                       5 \cdot 0 + 6 \cdot \frac{1}{3} \\
                & =    4
\end{align*}
\end{solution}

\ppart Consider the following identity (assume that all expectations
exist).
\begin{equation*}
\expect{R} =  \expcond{R}{C} \cdot \prob{C}  +
                        \expcond{R}{\overline{C}} \cdot \prob{\overline{C}}
\end{equation*}

Below is a proof of this identity.  Give a justification for each step
in this proof, appealing to either basic algebra or some rule of
probability given in the text.

We transform the right side into the left side.  All
summations are over $x \in \range{R}$.
\begin{align*}
\lefteqn{\prob{C} \cdot \expcond{R}{C} + \prob{\overline{C}} \cdot
  \expcond{R}{\overline{C}}} \\
& = \prob{C} \cdot \paren{ \sum_{x} x \cdot \prcond{R = x}{C} } +
\prob{\overline{C}} \cdot \paren{ \sum_{x} x \cdot \prcond{R =
  x}{\overline{C}} } \\
& = \prob{C} \cdot \paren{ \sum_{x} x \cdot \frac{\prob{[R = x] \intersect
    C}}{\prob{C}} } + \prob{\overline{C}} \cdot \paren{ \sum_{x} x \cdot
\frac{\prob{[R = x] \intersect \overline{C}}} {\prob{\overline{C}}} }\\
& = \sum_{x} x \cdot \paren{ \prob{[R = x] \intersect C} + \prob{[R = x]
  \intersect \overline{C}} } \\
& = \sum_{x} x \cdot \paren{ \prob{([R = x] \intersect C) \cup ([R = x]
  \intersect \overline{C})} } \\ & = \sum_{x} x \cdot \prob{R = x}\\
& = \expect{R}
\end{align*}

\begin{solution}
\begin{proof}
\begin{align*}
\lefteqn{\prob{C} \cdot \expcond{R}{C} + \prob{\overline{C}} \cdot
  \expcond{R}{\overline{C}}}\\ 
& = \prob{C} \cdot \paren{ \sum_{x} x \cdot \prcond{R = x}{C} } +
\prob{\overline{C}} \cdot \paren{ \sum_{x} x \cdot \prcond{R =
  x}{\overline{C}} }\\
        &\hspace{3in} \text{(def.\ of $\expcond{R}{D}$)}\\ 
& = \prob{C} \cdot \paren{ \sum_{x} x \cdot \frac{\prob{[R = x] \intersect
    C}}{\prob{C}} } + \prob{\overline{C}} \cdot \paren{ \sum_{x} x \cdot
\frac{\prob{[R = x] \intersect \overline{C}}} {\prob{\overline{C}}} } \\
       &\hspace{3in} \text{(def.\ of $\prcond{R = x}{D}$)}\\
& = \sum_{x} x \cdot \paren{ \prob{[R = x] \intersect C} + \prob{[R = x]
  \intersect \overline{C}} }\\
 &\hspace{3in} \text{(factoring)}\\
& = \sum_{x} x \cdot \paren{ \prob{([R = x] \intersect C) \cup ([R = x]
  \intersect \overline{C})} }\\
        &\hspace{3in} \text{(Sum Rule)}\\ 
& = \sum_{x} x \cdot \prob{R = x}\\
        &\hspace{3in} \text{(set identity $A=(A\intersect C)\union(A\intersect \bar{C})$)}\\ 
& = \expect{R}\\
        &\hspace{3in}\text{(by def.)}
\end{align*}
\end{proof}
\end{solution}

\eparts
\end{problem}
