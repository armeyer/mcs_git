\begin{problem}
\bparts
\ppart
Verify that the divisibilty relation on the set of nonnegative integers is
a weak  partial order.

\ppart What about the divisibilty relation on the set of integers?

\begin{solution}
Divisibility is not antisymmetric on the integers, since $n \divides -n$.
\end{solution}

\eparts

\end{problem}
