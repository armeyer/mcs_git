\begin{problem}

%% type: short-answer
%% title: Divisibility and Congruence

Which of the following statements are equivalent to each other?  Assume that $n > 0$ and $a$ and $b$ are
integers.
      
\begin{enumerate}
       
\item\ref{dc:choice1}
$a \equiv  b \pmod{n}$
       
\item\ref{dc:choice2}
$a$ = $b$
       
\item\ref{dc:choice3}
$\rem{a}{n} = \rem{b}{n}$
       
\item\ref{dc:choice4}
$n | (a - b)$
       
\item\ref{dc:choice5}
$a = b + nk$ for some integer $k$
       
\item\ref{dc:choice6}
($a - b$) is a multiple of $n$
       
\item\ref{dc:choice7}
$n | a$ or $n | b$
      
\end{enumerate}

%% <strong>Answer with a sequence of integers separated by
%% some spaces, for example, 
%% \begin{equation*}
%% 4 3 2 
%% \end{equation*}
%%  Don't use commas or
%% periods.

\begin{solution}
\ref{dc:choice1}, \ref{dc:choice3}, \ref{dc:choice4}, \ref{dc:choice5} and \ref{dc:choice6} are all equivalent to one another. (\ref{dc:choice2} and \ref{dc:choice7} are not.)
\end{solution}

\end{problem}
