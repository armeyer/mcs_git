\begin{problem}

Five basic properties of binary relations $R:A \to B$ are:

\begin{enumerate}

\item\label{assert:surj} $R$ is a surjection $[\geq 1\ \text{in}]$

\item\label{assert:inj} $R$ is an injection $[\leq 1\ \text{in}]$

\item\label{assert:func} $R$ is a function $[\leq 1\ \text{out}]$  %\geq typo corrected

\item\label{assert:total} $R$ is total $[\geq 1\ \text{out}]$

\item\label{assert:empty} $R$ is empty $[= 0\ \text{out}]$

%\item $R$ a bijection$[= 1\ \text{in,out}]$\hfill\examrule[1in]

\end{enumerate}

Below are some assertions about a relation $R$.  For each assertion,
write the numbers of all the properties above that the relation $R$
must have; write ``none'' if $R$ might not have any of these
properties.  For example, you should write ``\ref{assert:surj},
\ref{assert:total}'' next to the first assertion.

Variables $a,a_1,\dots$ range over $A$ and $b,b_1,\dots$ range over
$B$. You may assume $A$ and $B$ each have at least $2$ elements.

\bparts

\ppart $\forall a, b.\ \ a \mrel{R} b$.\hfill\examrule[0.4in]\underline{\ref{assert:surj},
    \ref{assert:total}}\examrule[0.4in]

\ppart $\QNOT(\forall a, b.\ \ a \mrel{R} b)$.\hfill\examrule[1.0in]
\begin{solution}
none
\end{solution}

\ppart $\forall a, b.\ \QNOT(a \mrel{R} b)$.\hfill\examrule[1.0in]
\begin{solution}
empty and therefore also injection and function: \ref{assert:empty}, \ref{assert:inj}, \ref{assert:func}
\end{solution}

\ppart $\forall a\, \exists b.\ \ a \mrel{R} b$.\hfill\examrule[1in]
\begin{solution}
total \ref{assert:total}
\end{solution}

\ppart $\forall b\, \exists a.\ \ a \mrel{R} b$.\hfill\examrule[1in]
\begin{solution}
surjection \ref{assert:surj}
\end{solution}

\ppart $R$ is a bijection.\hfill\examrule[1in]
\begin{solution}
all the properties \ref{assert:surj}, \ref{assert:inj}, \ref{assert:func}, \ref{assert:total}
\end{solution}

\ppart $\forall a\, \exists b_1.[a \mrel{R} b_1 \QAND (\forall b.\, a
\mrel{R} b \QIMPLIES b = b_1)]$.\hfill\examrule[1in]

\begin{solution}
total function  \ref{assert:func}, \ref{assert:total}
\end{solution}

%\ppart $\forall a,b.\ a \mrel{R} b \QOR a \neq b$.     %none

\iffalse
\ppart $\forall b_1,b_2,a.\ (a \mrel{R} b_1 \QAND a \mrel{R} b_2) \QIMPLIES b_1 = b_2$.
\begin{solution}
function \ref{assert:func}
\end{solution}

\ppart $\forall a_1,a_2,b.\ (a_1 \mrel{R} b \QAND a_2 \mrel{R} b)
\QIMPLIES a_1 = a_2$.\hfill\examrule[1in]
\begin{solution}
injection \ref{assert:inj}
\end{solution}
\fi

\ppart $\forall a_1,a_2,b_1,b_2.\ (a_1 \mrel{R} b_1\ \QAND\ a_2 \mrel{R}
  b_2\ \QAND\ a_1 \neq a_2) \QIMPLIES b_1 \neq b_2$.\hfill\examrule[1in]
\begin{solution}
injection \ref{assert:inj}
\end{solution}

\iffalse
\ppart $\forall a_1,a_2,b_1,b_2.\ (a_1\mrel{R}b_1 \QAND a_2\mrel{R}b_2
  \QAND b_1\neq b_2) \QIMPLIES a_1\neq a_2$.\hfill\examrule[1in]
\begin{solution}
function \ref{assert:func}
\end{solution}
\fi

\eparts

\end{problem}
