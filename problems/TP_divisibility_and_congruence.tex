\begin{problem}
List the numbers of all statements below that are \emph{equivalent} to
\[
a \equiv  b \pmod n,
\]
where $n > 1$ and $a$ and $b$ are integers.  \inbook{Briefly explain
  your reasoning.}
      
\exambox{2.0in}{0.5in}{0.0in}
\begin{solution}
\begin{center}
\fbox{ii, iv, vii, viii, ix}
\end{center}
\end{solution}

\begin{staffnotes}
Rubric: +3 for each equivalence listed, -2 for each non-equivalence
\end{staffnotes}

\begin{enumerate}[i)]
       
\item $2a \equiv 2b \pmod n$
            
\item $2a \equiv 2b \pmod{2n}$
            
\begin{solution}
equivalent
\end{solution}

\item
 $a^3 \equiv  b^3 \pmod n$

\item
 $\rem{a}{n} = \rem{b}{n}$

\begin{solution}
equivalent
\end{solution}

\item
 $\rem{n}{a} = \rem{n}{b}$

\item
 $\gcd(a,n) = \gcd(b,n)$

\item
 $\gcd(n,a-b) = n$

\begin{solution}
equivalent
\end{solution}

\item
 $(a - b)$ is a multiple of $n$

\begin{solution}
equivalent
\end{solution}

\item
 $\exists k \in \integers.\, a = b + nk$
       
\begin{solution}
equivalent
\end{solution}

\end{enumerate}

\end{problem}
