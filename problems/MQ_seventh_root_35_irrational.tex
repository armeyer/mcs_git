\begin{problem}\label{generprob}
   Prove that $\sqrt[7]{35}$ is irrational.

\begin{solution}
\begin{proof}
  Assume for the sake of contradiction that $\sqrt[7]{35}$ is rational.
  Under this assumption, there exist integers $a$ and $b$ such that
\[
\sqrt[7]{35} =  \frac{a}{b},
\]
where $a$ and $b$ have no common factor.  Now we prove that $a$ and $b$
have 5 as a common factor, a contradiction.

\begin{align}
\sqrt[7]{35} & = \frac{a}{b}, & \text{(by assumption)}\notag\\
35           & = \frac{a^{7}}{b^{7}}, & \text{(taking 7th powers)}\notag\\
35b^{7}      & = a^{7}.\label{35b7}
\end{align}
Since 5 is a factor of the left-hand side of~\eqref{35b7}, it is also a
factor of the right-hand side $a^{7}$.  By unique factorization, this
implies that 5 is a factor of $a$.

In particular, $a = 5c$ for some integer $c$.  Thus,
\begin{align}
35b^{7} & = (5c)^7 = 5^7 c^7,\notag\\
7b^{7}  & = 5^6c^{7} & \text{(dividing by 5)}.\label{7b7=56c7}
\end{align}
Since 5 is a factor of the right-hand side of~\eqref{7b7=56c7}, it is
also a factor of the left-hand side $7b^{7}$.  But 5 is not a factor
of 7, so by unique factorization it must be a factor of $b^7$ and
hence of $b$.
\end{proof}

\end{solution}

\end{problem}
