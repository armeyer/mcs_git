\begin{problem}
A flip of Coin 1 is $x$ times as likely to come up Heads as a flip of
Coin 2.  A biased random choice of one of these coins will be made,
where the probability of choosing Coin 1 is $w$ times that of Coin 2.
\iffalse The chosen coin is flipped and comes up Heads.\fi

\begin{editingnotes}
State the probs assumed > 0?
\end{editingnotes}

\bparts

\ppart Restate the information above as equations between
conditional probabilities involving the events
\begin{align*}
  C1 & \eqdef \text{Coin 1 was chosen},\\
  C2 & \eqdef \text{Coin 2 was chosen},\\
  H  & \eqdef \text{the chosen coin came up Heads}.
\end{align*}

\exambox{5.5in}{0.75in}{-0.2in}

\examspace[0.2in]

\begin{solution}
We are given that
\begin{equation}\label{HC1x}
\prcond{H}{C1} = x \prcond{H}{C2},
\end{equation}
and
\[
\pr{C1} = w \pr{C2},
\]
\end{solution}

\ppart State an inequality involving conditional probabilities of the
above events that formalizes the assertion ``Given that the chosen
coin came up Heads, the chosen coin is more likely to have been Coin 1
than Coin 2.''
\exambox{2.0in}{0.75in}{-0.2in}

\examspace[0.2in]

\begin{solution}

\begin{equation}%\label{REBE}
\prcond{C1}{H} >  \prcond{C2}{H}.
\end{equation}

\end{solution}

\ppart Prove that, given that the chosen coin came up Heads, the
chosen coin is more likely to have been Coin 1 than Coin 2 iff
\[
wx > 1.
\]

\begin{solution}
From~\eqref{HC1x} and the definition of conditional probability,
\[
\frac{\pr{H \QAND C1}}{\pr{C1}} = x \frac{\pr{H \QAND C2}}{\pr{C2}}.
\]
Substituting for $\pr{C1}$, we get
\[
\frac{\pr{H \QAND C1}}{w\pr{C2}} = x \frac{\pr{H \QAND  C2}}{\pr{C2}},
\]
which implies
\[
\pr{H \QAND C1} = wx\pr{H \QAND C2}.
\]
Dividing both sides of this inequality by $\pr{H}$, yields
\[
\prcond{C1}{H} = wx \prcond{C2}{H},
\]
so 
\[
\prcond{C1}{H} > \prcond{C2}{H}\quad  \QIFF\quad  wx > 1
\]

\end{solution}

\eparts

\end{problem}
