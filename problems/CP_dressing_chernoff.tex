\begin{problem}
%\section{Getting Dressed}

Sometimes I forget a few items when I leave the house in the morning.
For example, here are probabilities that I forget various pieces of
footwear:
%
\[
\begin{array}{p{2in}l}
left sock & 0.2 \\
right sock & 0.1 \\
left shoe & 0.1 \\
right shoe & 0.3
\end{array}
\]
%
\bparts

\ppart %\item[a.]

Let $X$ be the number of these that I forget.  What is $\expect{X}$?

\examspace[1in]

\begin{solution}

%By Theorem~\ref{th:num-events}, 

$X$ is the sum of the indicator variables for forgetting each item,
  and the expectation of an indicator is the probability of the
  expected event, so the expected number of events that happen is the
  sum of the event probabilities:
\[
\expect{X} = 0.2 + 0.1 + 0.1 + 0.3 = 0.7
\]
\end{solution}

\ppart  %\item[b.]

Give a tight upper bound on the probability that I forget one or more
items when no independence assumption is made about forgetting
different items.

\examspace[1in]

\begin{solution}
By the Union Bound, the probability that I forget something is at
most the sum of the probabilities of forgetting each single item, namely,
\[
0.1 + 0.1 + 0.2 + 0.3 = 0.7
\]
In this case, Markov's Bound yields the same answer.

\end{solution}

\ppart  %\item[c.]

Use the \idx{Markov Bound} to derive an upper bound on the probability
that I forget 3 or more items.

\examspace[1.5in]

\begin{solution}
\[
\prob{X \geq 3} \leq \frac{\expect{X}}{3} = \frac{7}{30} < 0.234\ .
\]

In this case, a different simple argument yields a better bound.
Namely, if $[X \geq 3]$, then at least one of right sock and
left shoe must have been forgotten, so
\begin{align*}
\prob{X \geq 3}
  & \leq \prob{\text{forgot right sock or left shoe}}\\
  & \leq \prob{\text{forgot right sock}} + {\text{forgot right left
    shoe}}\\
  & = 0.1 + 0.1 = 0.2\ .
\end{align*}
\end{solution}

\ppart  %\item[d.]

Now suppose that I forget each item of footwear independently.  Use
the \idx{Chebyshev Bound} to derive an upper bound on the probability
that I forget two or more items.

\examspace[1.5in]

\begin{solution}
 Let $X_1$ be the event I bring my left sock, $X_2$ my right sock,
 $X_3$ my left shoe, and $X_4$ my right shoe.  Then
\[
\expect{X_1} = 0.2, \expect{X_2} = 0.1, \expect{X_3} = 0.1, \expect{X_4} = 0.3.
\]
Moreover, since the $X_i$ are \idx{indicator} (0-1 valued) random
variables, we have
\begin{align*}
\variance{X_1} & = 0.2(1-.2) = 0.16, \variance{X_2} = 0.1(1-.1) = 0.09,\\
\variance{X_3} & = 0.1(1-.1) = 0.09, \variance{X_4} = 0.3(1-.3) = 0.21.
\end{align*}

Let $X = \sum_{i=1}^4 X_i$. Then
\[
\expect{X} = \sum_{i=1}^4 \expect{X_i} = 0.2 + 0.1 + 0.1 + 0.3 = 0.7.
\]
Since the $X_i$ are independent,
\[
\variance{X} = \sum_{i=1}^4 \variance{X_i} = 0.16 + 0.09 + 0.09 + 0.21 = 0.55.
\]

Now by Chebyshev's Inequality,
\begin{align*}
\prob{X \geq 2} & \leq \prob{\abs{X-.7} \geq 1.3} = \prob{\abs{X-\expect{X}} \geq 1.3}\\
                & \leq \frac{\variance{X}}{1.3^2} = \frac{.55}{1.3^2}
\leq 0.326\ .
\end{align*}
\end{solution}

\ppart %\item[e.]

Use \idx{Murphy's Law}\inbook{, Theorem~\bref{th:murphy},} to derive a
lower bound on the probability that I forget one or more items.

\examspace[1in]

\begin{solution}
Plugging into Theorem~\bref{th:murphy}, the probability that I forget
one or more items is at least
\[
1 - e^{-\expect{X}} = 1 - e^{-0.7} = 0.503\dots
\]
\end{solution}

\iffalse
\ppart %\item[f.]

I'm supposed to remember many other items, of course: clothing, watch,
backpack, notebook, pencil, kleenex, ID, keys, etc.  Let $X$ be the
total number of items I remember.  Suppose I remember items mutually
independently and $\expect{X} = 36$.

Give an upper bound on the probability that I remember 18 or fewer
items.

\examspace[1in]

\begin{solution}

By the first Chernoff inequality:
%
\begin{align*}
\prob{X \leq 18}
    & = \prob{X \leq (1 - 1/2) \expect{X}} \\
    & \leq e^{-(1/2)^2 36 / 2} \\
    & = e^{-9/2}
\end{align*}
\end{solution}
\fi

\ppart %\item[g.]
 
I'm supposed to remember many other items, of course: clothing, watch,
backpack, notebook, pencil, kleenex, ID, keys, etc.  Let $X$ be the
total number of items I remember.  Suppose I remember items mutually
independently and $\expect{X} = 36$.  Use Chernoff's Bound to give an
upper bound on the probability that I remember 48 or more items.

\examspace[2in]

\begin{solution}

By the Chernoff Bound (with $\beta(c) \eqdef  c\ln c -c +1$),
\[
\prob{X \geq 48} = \prob{X \geq (1 + 1/3) \expect{X}} \leq e^{-\beta(4/3)\cdot 36} \approx 0.1638
\]
\end{solution}

\ppart
%\item[h.]

Give an upper bound on the probability that I remember $108$ or more
items.

\begin{solution}
By the Chernoff Bound,
\[
\prob{X \geq 108}
   = \prob{X \geq 3 \cdot \expect{X}}
   \leq e^{-\beta(3) \cdot 36}
   \leq e^{-46}
   \approx 1.05 \times 10^{-20}.
\]
\end{solution}

\eparts

\end{problem}
