\begin{problem}
Let $A,B,C$ be events.  For each of the following statements, prove it
or give a counterexample.

\bparts

\ppart\label{AindBAbar}
If $A$ is independent of $B$, then $A$ is also independent of
$\bar{B}$.

\begin{solution}
\True.

To prove it, suppose $A$ is independent of $B$; that is,
$\prob{A\intersect B}=\prob{A}\prob{B}$.  Then
\begin{align*}
\pr{A \intersect \setcomp{B}}
  &  = \pr{A} - \pr{A \intersect B}
       & \text{(difference rule)}\\
  & =  \pr{A} - \pr{A} \cdot \pr{B}
       & \text{(independence of $A$ and $B$)}\\
  & =  \pr{A} (1 - \pr{B})\\
  & =  \pr{A} \cdot \pr{\setcomp{B}}. & \text{(complement rule)}
\end{align*}
\end{solution}


\ppart\label{AindBindCthenABC}
If $A$ is independent of $B$, and $A$ is independent of $C$, 
then $A$ is independent of $B \intersect C$.

\hint Choose $A,B,C$ pairwise but not 3-way independent.

\begin{solution}
\False.

Let $A,B,C$ be any set of pairwise but not 3-way independent events.

So $A$ is independent of $B$ and of $C$ by pairwise independence, but
\begin{align*}
\prcond{A}{B \intersect C}
 & \eqdef \frac{\pr{A \intersect (B \intersect  C)}}{\pr{B \intersect  C}}\\
 & \neq \frac{\pr{A}\cdot \pr{B} \cdot \pr{C}}{\pr{B \intersect  C}}
       & \text{($A,B,C$ not 3-way independent)}\\
 & = \frac{\pr{A}\cdot \pr{B} \cdot \pr{C}}{\pr{B} \cdot \pr{C}}
       & \text{($B,C$ pairwise independent)}\\
 & = \pr{A},
\end{align*}
so $A$ is not independent of $B \intersect C$.
\end{solution}

\ppart 
If $A$ is independent of $B$, and $A$ is independent of $C$,
then $A$ is independent of $B\union C$.

\hint Part~\eqref{AindBindCthenABC}.

\begin{solution}
\False.

As in part~\eqref{AindBindCthenABC}, let $A,B,C$ be any set of
pairwise but not 3-way independent events.  So $A$ is independent of
both $B$ and $C$ by pairwise independence.

Now by part~\eqref{AindBAbar}, $A$ is independent of $B \union C$ iff
it is independent of $\bar{B \union C}$.  That is, $A$ must be
independent of $\bar{B} \intersect \bar{C}$.

But by part~\eqref{AindBAbar} again $A$ $\bar{B}$, and $\bar{C}$ are
also pairwise but not 3-way independent, and we saw in
part~\eqref{AindBindCthenABC} that in this case, $A$ is not
independent of $\bar{B} \intersect \bar{C}$.

It is also easy to give a concrete counterexample.  Let the
probability space be the integer interval $\Zintv{1}{4}$ with uniform
probability, and let
\[
A \eqdef [\text{Even}], \qquad B \eqdef [\leq 2], \qquad C \eqdef \set{2,3}.
\]
Then
\[
\pr{A} = \frac12 = \prcond{A}{B} = \prcond{A}{C},
\]
so $A$ is independent of $B$ and of $C$.

But
\[
\prcond{A}{B\union C} = \prcond{[\text{even}]}{\Zintv{1}{3}} = \frac13 \neq \frac12 = \pr{A}, 
\]
so $A$ is not independent of $B\union C$.
\end{solution}

\ppart 
If $A$ is independent of $B$, and $A$ is independent of $C$, and $A$
is independent of $B\intersect C$, 
then $A$ is independent of $B\union C$.

\begin{solution}
\True.

To prove it, suppose $A$ is independent of each of $B$, $C$ and
$B\intersect C$, so
\begin{align}
\prob{A\intersect B} &= \prob{A}\prob{B} \label{AintBAB}\\
\prob{A\intersect C} &= \prob{A}\prob{C}  \label{AintCAC}\\
\prob{A\intersect (B\intersect C)}
&= \prob{A}\prob{B\intersect C}. \label{AintBintCABC}
\end{align}
Then, 
\begin{align*}
\lefteqn{\prob{A\intersect(B\union C)}}\\
  & = \prob{(A\intersect B) \union (A\intersect C)}\\
  & = \prob{A\intersect B} + \prob{A\intersect C}\\
  &\qquad       - \prob{(A\intersect B)\intersect(A\intersect C)}
            & \text{(inclusion-exclusion)}\\
  & = \prob{A\intersect B} + \prob{A\intersect C}
          - \prob{A\intersect B\intersect C}\\
  & = \prob{A}\prob{B} + \prob{A}\prob{C}
          - \prob{A}\prob{B\intersect C}
            & \text{(by \eqref{AintBAB}, \eqref{AintCAC}, \eqref{AintBintCABC})}\\
  & = \prob{A}(\prob{B} + \prob{C} - \prob{B\intersect C})\\
  & = \prob{A}\prob{B \union C}.
       & \text{(inclusion-exclusion)}
\end{align*}

\end{solution}

\eparts
