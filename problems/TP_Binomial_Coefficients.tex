\begin{problem}
Find the coefficient of $x^{4}$ in the following expressions.

\bparts

\ppart
$(x+1)^{9}$

\begin{solution}
126

$= \binom{9}{4} 1^3 1^{9-3}$
\end{solution}

\ppart
$(3x+2)^{6}$ 

\begin{solution}
4860

$= \binom{6}}{4}3^4 2^{6-4}$
\end{solution}

\eparts

\end{problem}
