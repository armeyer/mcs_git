\begin{problem}
  There are four isomorphisms between these two graphs. List them.
  (Partial credit for listing at least two.)

\begin{figure}[h]
\graphic[height=2in]{isomorphism_a}
\end{figure}

\vspace{10 mm}

\begin{figure}[h]
\graphic[height=2in]{isomorphism_b}
\end{figure}

\begin{solution}
These are the vertex correspondences for the four isomorphisms: \\
1B, 2C, 3D, 4A \\
1B, 2A, 3D, 4C \\
1D, 2C, 3B, 4A \\
1D, 2A, 3B, 4C
\end{solution}

\end{problem}
