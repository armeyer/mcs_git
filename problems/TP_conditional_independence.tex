\begin{problem}
Give examples of event $A,B,E$ such that

\begin{staffnotes}
For online, maybe make into multiple choice.
\end{staffnotes}
\bparts

\ppart $A$ and $B$ are independent, and are also conditionally
independent given $E$, but are not conditionally independent given $\setcomp{E}$.
That is,
\begin{align*}
  \pr{A \intersect B} & = \pr{A}\pr{B},\\
  \prcond{A \intersect B}{E} & = \prcond{A}{E} \prcond{B}{E},\\
  \prcond{A \intersect B}{\setcomp{E}} & \neq \prcond{A}{\setcomp{E}} \prcond{B}{\setcomp{E}}.
\end{align*}

\hint Let $\sspace = \set{1,2,3,4}$.

\begin{staffnotes}
There may be simpler examples.

Also, I believe the example below could be tweaked so none of the
probabilities involved are zero or one; if so, one could ask for such.
\end{staffnotes}

\begin{solution}
Over the space $\set{1,2,3,4}$, let
\begin{align*}
  A & \eqdef \set{1,2}\\
  B & \eqdef \set{2,3}\\
  E & \eqdef \set{4}.
\end{align*}
\end{solution}

\ppart $A$ and $B$ are conditionally independent given $E$, or given
$\setcomp{E}$, but are not independent.  That is,
\begin{align*}
  \prcond{A \intersect B}{E} & = \prcond{A}{E} \prcond{B}{E},\\
  \prcond{A \intersect B}{\setcomp{E}} & = \prcond{A}{\setcomp{E}} \prcond{B}{\setcomp{E}},\\
  \pr{A \intersect B} & \neq \pr{A}\pr{B}.
\end{align*}

\hint Let $\sspace = \set{1,2,3,4,5}$.

\begin{staffnotes}
There may be simpler examples.

Also, I believe the examples below could be tweaked so none of the
probabilities involved are zero or one; if so, one could ask for such.
\end{staffnotes}

\begin{solution}
Over the space $\set{1,2,3,4,5}$, let
\begin{align*}
  A & \eqdef \set{1,2}\\
  B & \eqdef \set{2,3}\\
  E & \eqdef \set{5}.
\end{align*}

An alternative example is
\begin{align*}
  A & \eqdef \set{1}\\
  B & \eqdef \set{1,2}\\
  E & \eqdef \set{3,4,5}.
\end{align*}
\end{solution}

\eparts

\end{problem}
