\begin{problem}

\begin{problemparts}

\problempart Show two planar drawings of a \textbf{single} planar
graph, such that one of the drawings corresponds to a planar embedding
in which at least one discrete face's length is different from the
lengths of any of the discrete faces in the embedding associated with
the other drawing.

\examspace[4in]

\begin{solution}
The pair of planar drawings below meet the given requirements.  The
planar embedding corresponding to Figure~\ref{fig:planar-1-sol}
includes the discrete face $abdeca$, which has length 5.  The longest
discrete face in the planar embedding corresponding to
Figure~\ref{fig:planar-2-sol} has length 4.

\begin{figure}[h]
\graphic{MQ_planar_drawing_1}
\caption{First planar drawing.\label{fig:planar-1-sol}}
\end{figure}
\examspace[2in]
\begin{figure}[h]
\graphic{MQ_planar_drawing_2}
\caption{Second planar drawing.\label{fig:planar-2-sol}}
\end{figure}
\end{solution}

\problempart
List the discrete faces in each of these two planar embeddings. 

\begin{solution}
For the particular pair shown in Figures~\ref{fig:planar-1-sol}
and~\ref{fig:planar-2-sol}, the embeddings are as follows:

\begin{center}
\begin{tabular}{|l|l|}
  \hline
  Figure~\ref{fig:planar-1-sol} & Figure~\ref{fig:planar-2-sol} \\ \hline
  $abca$ & $abca$\\
  $bcdb$ & $bcedb$\\
  $cdec$ & $cdec$ \\
  $abdeca$ & $abdca$ \\ \hline
\end{tabular}
\end{center}
\end{solution}
