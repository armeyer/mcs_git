\begin{problem}
Give a combinatorial proof that
\[
\binom{n}{k} = \binom{n}{n-k}
\]
(It is possible to answer with two short sentences.)

\examspace[2.0in]

\begin{solution}
$\binom{n}{k}$ is the number of ways to choose $k$ out of $n$ elements
  to keep.  $\binom{n}{n-k}$ is the number of ways to choose the $n-k$
  elements not to keep.
\end{solution}

\end{problem}
