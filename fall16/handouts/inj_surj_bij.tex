\subsection*{A function $f: A\rightarrow B$ (``from A to B'') is:}
\begin{itemize}
%\item a \term{function} if every element of $A$ is assigned to at most one element of $B$,
%
%\item
%\term{total} when every element of~$A$ is assigned to at least one element of~$B$. More
%concisely, $R$ is total iff $A = R^{-1}(B)$.

\item
\term{surjective} if every element of~$B$ is assigned to at least one
element of~$A$. More concisely, $f$ is surjective iff $f(A) = B$ (that is, if the range of $f$ is the codomain of $f$).

\item
\term{injective} if every element of~$B$ is mapped \emph{at most once}, and

\item
\term{bijective} if it is surjective and injective.
\end{itemize}

\subsection*{Given any two sets $A$ and $B$, we say that:}
\begin{itemize}
\item $|A| \geq |B|$ iff there is a surjection from A to B
\item $|A| \leq |B|$ iff there is an injection from A to B
\item $|A| = |B|$ iff there is a bijection between A and B

\end{itemize}
