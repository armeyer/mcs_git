\begin{problem} \mbox{}

Indicate \True\ or \False\ for the following statements about finite
\textbf{simple graphs} $G$.

\bparts

\ppart  $G$ has a spanning tree.  \hfill \True\ \qquad
  \False\

\begin{solution}
\False.  Any disconnected graph is a counterexample.
\end{solution}

\ppart $\card{\vertices{G}} = O(\card{\edges{G}})$ for connected $G$.  \hfill \True\ \qquad
  \False\

\begin{solution}
\True.
\end{solution}

\inbook{
\ppart $\card{\edges{G}} = O(\card{\vertices{G}})$.  \hfill \True\ \qquad
  \False\

\begin{solution}
\False.  $\card{\vertices{K_n}} = n = o(n^2)$, but
  $\card{\edges{K_n}} = \Theta(n^2)$.

\end{solution}
}

\ppart $\chi(G) \leq \max\set{\degr{v} \suchthat
  v \in \vertices{G}}$.\footnote{$\chi(G)$ is the chromatic number of $G$.}  \hfill \True\ \qquad
  \False\

\begin{solution}
\False.  $\chi(K_n) = n > n-1 = \text{ max degree of } v \in \vertices{K_n}$
\end{solution}

\ppart $\card{\vertices{G}} = O(\chi(G))$.  \hfill \True\ \qquad
  \False\ 

\begin{solution}
\False.

There are arbitrarily large graphs, for example trees, with $\chi = 2$.
\end{solution}

\eparts

\end{problem}
