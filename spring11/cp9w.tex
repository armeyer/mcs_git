\section*{Asymptotic Notations}

\iffalse

\begin{center}Stirling's Formula\end{center}
\[
n! \sim \paren{\frac{n}{e}}^n \sqrt{2 \pi n},
\]

More precisely,
\begin{fact*}[Stirling's Approximation]
\[
\sqrt{2 \pi n} \paren{\frac{n}{e}}^n e^{1/(12n+1)} \leq n! \leq
\sqrt{2 \pi n} \paren{\frac{n}{e}}^n e^{1/12n}.
\]
\end{fact*}
\fi

Let $f,g$ be functions from $\reals$ to $\reals$.

\begin{itemize}

\item $f$ is \emph{asymptotically equal} to $g$:\quad $f(x) \sim g(x)
  \qiff \lim_{x \rightarrow \infty} f(x)/g(x) = 1$.

\item $f$ is \emph{asymptotically smaller} than $g$:\quad $f(x) = o(g(x))
  \qiff \lim_{x \rightarrow \infty} f(x)/g(x) = 0$.

\item for $f,g$ nonnegative,\quad $f = O(g)
\qiff
\limsup_{x \rightarrow \infty} f(x)/g(x) < \infty$,
where

$\limsup_{x \rightarrow \infty} h(x) \eqdef \lim_{x \rightarrow \infty}
\text{lub}_{y \geq x} h(y)$.

An alternative, equivalent, definition is
\[
f = O(g)
\qiff \exists c, x_0 \in \reals^+\, \forall x \geq x_0.\ f(x) \leq c g(x).
\]

\item Finally, $f = \Theta(g) \qiff f=O(g) \QAND g=O(f)$.

\end{itemize}
