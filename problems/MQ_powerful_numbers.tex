\begin{problem}

\bparts\mbox{}

\ppart What is the value of $\phi(175)$, where $\phi$ is Euler's function?

\examspace[1.5in]

\begin{solution}
  Noting that $175 = 5^2 \cdot 7$.  It follows
  that $\phi(175) = (5^2 - 5^1)(7-1) = 20 \cdot 6 = 120$.
\end{solution}

\iffalse
\ppart Call a number from 0 to 174 \emph{powerful} iff some positive
power of the number is congruent to 1 modulo 175.  How many powerful
numbers are there?  Explain.

\examspace[1.5in]

\begin{solution}
\[
120.
\]

  Note that $x^k \equiv 1 \pmod n$ for some $k$ iff $x$ has an inverse ($x^{k-1}$)
  modulo $n$ iff $x$ is relatively prime to $n$.  So being powerful is
  equivalent to being relatively prime to 175.  There are $\phi(175) =
  (5^2 - 5^1)(7-1) = 20 \cdot 6 = 120$ numbers from 0 to 174 that are
  relatively prime to 175.
\end{solution}
\fi

\ppart\label{12to482rem}
What is the remainder of $(-12)^{482}$ divided by 175?

\examspace[1.0in]

\begin{solution}
  \textbf{144}.

  Since -12 and 175 are relatively prime, we have by Euler's
  Theorem that $(-12)^{\phi(175)} \equiv 1 \pmod {175}$, and so
a  \[
  (-12)^{482} = \paren{(-12)^{120}}^{4} \cdot (-12)^{2}
              \equiv 1^{4} \cdot (144)
              \equiv 144 \pmod {175}.
  \]
\end{solution}

\ppart Explain why $(-12)^{482}$ has a multiplicative inverse modulo 175.

\examspace[2in]

\begin{solution}
  A number has a multiplicative inverse modulo 175 iff it is
  relatively prime to 175.  But the only prime divisors of 175 are 7
  and 5, and for any $k \in \nngint$, the only prime divisors of
  $(-12)^k$ are the prime divisors of 12, namely 2 and 3, so $(-12)^k$
  is relatively prime to 175.

  Alternatively, using part~\eqref{12to482rem}, we just need to show
  that $\gcd(175,144) = 1$, which can be verified by the Euclidean
  algorithm:
\begin{align*}
\gcd(175,144) & = \gcd(144,31) = \gcd(31,20)\\
              & = \gcd(20,11) = \gcd(11,9)\\
              & = \gcd(9,2) = \gcd(2,1) = 1.
\end{align*}

\end{solution}
