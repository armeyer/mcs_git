\begin{problem}
Evaluate the following double sums and products!

\begin{enumerate}
\item $\prod_{k=1}^n 2^{2*k-1}$
\item $\sum_{i=0}^n \sum_{j=0}^m 3^{i+j}$
\item $\sum_{i=1}^n \sum_{j=1}^m (i+j)$
\end{enumerate}

\begin{solution}

\begin{align*}
 \prod_{k=1}^n 2^{2*k-1}
    &= 2^{\sum_{k=1}^n 2*k-1}\\
    &= 2^{n*(2*n-1+1)/2} \\
    &= 2^{n^2}
\end{align*}

\begin{align*}
\sum_{i=0}^n \sum_{j=0}^m 3^{i+j}
	&= (\sum_{i=0}^n 3^i)*(\sum_{j=0}^m 3^j)\\
	&= \frac{3^{i+1}-1}{2}*\frac{3^{j+1}-1}{2}
\end{align*}

\begin{align*}
\sum_{i=1}^n \sum_{j=1}^n (i+j)
	&= \sum_{i=1}^n n*i + n*(n+1)/2 \\
	&= n*n*(n+1)/2 + n*n*(n+1)/2 \\
	&= n^2*(n+1)
\end{align*}
\end{solution}

\end{problem}
