\begin{problem}
You've seen this neat trick for evaluating a geometric sum:
%
\begin{align*}
S & = 1 + z + z^2 + \cdots + z^n \\
zS & = z + z^2 + \cdots + z^n + z^{n+1} \\
S - z S & = 1 - z^{n+1} \\
S & = \frac{1-z^{n+1}}{1 - z} \text{ (where } z \neq 1)
\end{align*}

Use the same approach to find a closed-form expression for this sum:
%
\begin{align*}
T & = 1 z + 2 z^2 + 3 z^3 + \cdots + n z^n
\end{align*}

\begin{solution}
\begin{align*}
z T & = 1 z^2 + 2 z^3 + 3 z^4 + \cdots + n z^{n+1} \\
T - zT & = z + z^2 + z^3 + \cdots + z^n - n z^{n+1} \\
       & = \frac{1 - z^{n+1}}{1 - z} - 1 - n z^{n+1} \text{ (where } z \neq 1)\\
T & = \frac{1 - z^{n+1}}{(1 - z)^2} - \frac{1 + n z^{n+1}}{1 - z} \text{ (where } z \neq 1)
\end{align*}

\end{solution}

\end{problem}
