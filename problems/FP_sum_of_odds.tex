\begin{problem}


Prove by induction that
\begin{equation}\label{oddsumn2}
1 + 3 + 5 +\dots+ (2n+1)  = (n+1)^2.
\end{equation}

\begin{solution}
We prove that~\eqref{oddsumn2} holds for all $n \geq 0$ by induction,
with~\eqref{oddsumn2} as the induction hypothesis.

\inductioncase{Base case}: $(n = 0)$.  The equation becomes $1 =
(0+1)^2$, which is true, proving the base case.

\inductioncase{Inductive step}:
We may assume the induction hypothesis~\eqref{oddsumn2}.

Add $2(n+1) + 1$ to both sides of the equation.  The left-hand side is
now the sum of odd numbers up to $2(n+1)+1$.  The right-hand side becomes
\begin{equation}\label{rightn112}
(n+1)^2 + 2(n+1) + 1.
\end{equation}
To prove $P(n+1)$, we need only show that~\eqref{rightn112} equals
$((n+1)+1)^2$, that is $(n+2)^2$.  This follows by elementary algebra:

\[
(n+1)^2 + 2(n+1) + 1 
  = (n^2+2n + 1) + 2n +3
  = n^2 + 4n + 4
  = (n+2)^2
\]
This completes the proof of the induction step.

We conclude by induction that $P(n)$ is true for all $n\geq 0$.
\end{solution}

\end{problem}
