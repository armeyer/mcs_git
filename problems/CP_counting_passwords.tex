\begin{problem}
A password consists of 8 characters that allow only letters and
digits.  The rules for constructing the passwords are:

\begin{itemize}

\item Only 8 characters allowed

\item Only lower-case letters and digits

\item First character must be a letter

\item There should be at least one digit

\end{itemize}

Let $P$ be the set of all possible passwords.

\bparts

\ppart Express $P$ in terms of
%
\begin{align*}
\mathcal{A} & = \set{ a, b, c, \dots, z} \\
\mathcal{D} & = \set{ 0, 1, 2, \dots, 9}
\end{align*}
%
using unions ($\union$) and set products ($\times$).

\begin{solution}
\[
P = (A \times (A \union D )^7) - A^8 
\]

\end{solution}

\ppart Compute $\card{P}$, the number of different password,
using the sum and product rules.

\begin{solution}
\begin{align*}
\card{P}
  & = \Card{(A \times (A \union D)^7) - A^8 } \\
  & = \card{A} \cdot \card{A \union D}^7 - \card{A}^8 {Product Rule} \\
  & = \card{A} \cdot (\card{A} + \card{D})^7 - \card{A}^8  & \text{Sum Rule} \\
  & = 26 \cdot (26 + 10)^7 - 26^8
\end{align*}

\end{solution}

\eparts

\end{problem}
