\begin{problem}
Your class has a textbook and a final exam.  Let $P$, $Q$ and $R$ be
the following propositions:
\begin{description}
\item $P \eqdef$ You get an A on the final exam.
\item $Q \eqdef$ You do every exercise in the book.
\item $R \eqdef$ You get an A in the class.
\end{description}

Translate following assertions into propositional formulas using $P$,
$Q$, $R$ and the propositional connectives $\QAND, \QNOT, \QIMPLIES$.

\bparts

\ppart You get an A in the class, but you do not do every exercise in
the book.

\instatements{\vspace{0.5in}}
%\examspace[0.5in]

\begin{solution}
$R \QAND \QNOT(Q)$.  Alternatively, $\QNOT(R \QIMPLIES Q)$.
\end{solution}

\ppart
You get an A on the final, you do every exercise in the
book, and you get an A in the class.

\instatements{\vspace{0.5in}}

\begin{solution}
$P \QAND Q \QAND R$
\end{solution}

\ppart
To get an A in the class, it is necessary for you to get
an A on the final.

\instatements{\vspace{0.5in}}

\begin{solution}
$R \QIMPLIES P$.
\end{solution}

\ppart
You get an A on the final, but you don't do every exercise
in this book; nevertheless, you get an A in this class.

\instatements{\vspace{0.5in}}

\begin{solution}
$P \QAND \QNOT(Q) \QAND R$.   Alternatively, $\QNOT([P \QAND R] \QIMPLIES Q)$.
\end{solution}

\eparts

\end{problem}
