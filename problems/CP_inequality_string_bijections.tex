\begin{problem} \mbox{}
\bparts

\ppart Let $\mathcal{S}_{n,k}$ be the possible nonnegative integer solutions to
the inequality
\begin{equation}\label{intsoln}
x_1 + x_2 + \cdots + x_k \leq n.
\end{equation}
That is
\[
\mathcal{S}_{n,k} \eqdef \set{(x_1,x_2,\dots,x_k) \in \nngint^k \suchthat
  \text{\eqref{intsoln} is true}}.
\]
Describe a bijection between $\mathcal{S}_{n,k}$ and the set of binary
strings with $n$ zeroes and $k$ ones.

\begin{solution}
The notation $\mathtt{0}^x$ indicates a length $x$ string of
$\mathtt{0}$'s.
\[
(x_1,x_2,\dots,x_k) \corresp
\mathtt{0}^{x_1}\mathtt{10}^{x_2}\mathtt{1} \dots \mathtt{0}^{x_k}\mtt{10}^{n-s},
\]
where $s \eqdef \sum_{i=1}^k x_i$.
\end{solution}

\ppart Let $\mathcal{L}_{n,k}$ be the length $k$ weakly increasing
sequences of nonnegative integers $\leq n$.  That is
\[
\mathcal{L}_{n,k} \eqdef \set{(y_1,y_2,\dots,y_k) \in \nngint^k \suchthat y_1
\leq y_2 \leq \dots \leq y_k \leq n}.
\]

Describe a bijection between $\mathcal{L}_{n,k}$ and $\mathcal{S}_{n,k}$.

\begin{solution}
$(y_1,y_2,\dots,y_k) \corresp (y_1,\ y_2-y_1,\ y_3-y_2,\ \dots,\ y_k-y_{k-1})$.

In the other direction,
\[
(x_1,x_2,\dots, x_k) \corresp (x_1,\ x_1+x_2,\ x_1+x_2+x_3,\ \dots,\ \sum_{i=1}^k x_i).
\]
\end{solution}

\eparts
\end{problem}
