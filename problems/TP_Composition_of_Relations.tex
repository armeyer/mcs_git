\begin{problem}
Let             
\begin{align*}
A & \eqdef \set{1,2,3}\\
B & \eqdef \set{4,5,6}\\
R & \eqdef \set{(1,4),(1,5),(2,5),(3,6)}\\
S & \eqdef \set{(4,5),(4,6),(5,4)}.\\
\end{align*}
Note that $R$ is a relation from $A$ to $B$ and $S$ is a relation from
$B$ to $B$.

List the pairs in each of the relations below.
 %   (For example, to list the pairs in $S$, you should type:
     
%\begin{equation*}
%(4 5) (4 6) (5 4)
%\end{equation*}

%     without commas or outer parens.)
\bparts

\ppart $S \compose R$.

\begin{solution}
$(1, 5), (1, 6), (1, 4), (2, 4).$
\end{solution}

\ppart $S \circ  S$.

\begin{solution}
$(4, 4), (5, 5), (5, 6).$
\end{solution}

\ppart $S^{-1} \circ  R$.

\begin{solution}
$(1, 5), (1, 4),(2, 4),(3, 4).$
\end{solution}

\eparts

\end{problem}
