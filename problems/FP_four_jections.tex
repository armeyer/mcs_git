\begin{problem}
For each pair $(A,B)$ of sets below, indicate \emph{exactly} which of the
following kinds of binary relations from $A$ to $B$ are possible.  The
kinds of relations are:

\begin{center}
\begin{tabular}{rl}
 TI:   &  Total Injection\\
 TIF:  &  Total injective Function\\
 SF:   &  Surjective Function\\
 TSF:  &  Total Surjective Function
\end{tabular}
\end{center}
      
Notation:
\begin{align*}
\strings{X} & \eqdef \text{finite strings}\ x_0x_1\dots x_n,\\
\infstrings{X} & \eqdef \text{infinite strings}\ x_0x_1\dots x_n \dots,
\end{align*}
where $x_i \in X$ for all $i \geq 0$.

The pairs are:

\[\begin{array}{c|c|r}
A & B \\
\hline \integers &  \nngint & \examrule\\
%\power(\rationals) &  \nngint & \examrule\\
\rationals &  \finbin & \examrule\\
\reals \times \reals &  \reals & \examrule\\
\binw & \ \infstrings{\set{2,3,4}} & \examrule\\
\finbin & \binw & \examrule\\
%\set{n \in \nngint \suchthat n \leq 2^{100}} & \power(\set{0,1,...,100}) & \examrule\\
\power(\finbin) & \binw & \examrule\\
\emptyset &  \integers & \examrule\\
\nngint &  \emptyset & \examrule
\end{array}\]

\begin{solution}
\begin{enumerate}[i]
\item ALL
\item ALL
\item ALL
\item ALL
\item TI, TIF
\item ALL
\item TI, TIF
\item SF
\end{enumerate}


\end{solution}

\end{problem}
