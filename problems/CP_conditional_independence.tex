\begin{problem}
Let $A,B,C,D$ be events.  Describe counterexamples showing that the
following claims are false.

\bparts

\ppart

\begin{falseclm*}
If $A$ and $B$ are independent given $C$, and are also
independent given $D$, then $A$ and $B$ are independent given $C
\union D$.
\end{falseclm*}

\begin{staffnotes}
After a few minutes, OK to give \hint $C=A$, $D=\bar{A}$.
\end{staffnotes}

\begin{solution}
Notice that $A$ is certain given $A$, so everything is independent of
$A$ given $A$, and similarly everything is independent of $A$ given
$\bar{A}$.  But an event is independent of $A$ given $A \union
\bar{A}$ iff it simply is independent $A$.  So $C=A$, $D=\bar{A}$, and
$B$ any event that is not independent of $A$ provides a
counterexample.

\begin{staffnotes}
You could ask students to work this out concretely:

Let $B$ and $C$ be $A$ and $D$ be $\bar{A}$.  Then
\begin{align*}
\prcond{A \intersect B}{C}
  & = \pr{A \intersect A \intersect A}/\pr{A}\\
  & = 1\\
  & = \prcond{A}{A} \cdot \prcond{A}{A}\\
  & = \pr{A}{C} \cdot \pr{A}{B},
\end{align*}
so $A$ and $B$ are independent given $C$.  Also,
\[
\prcond{A \intersect B}{D} = 0 = \prcond{B}{D},
\]
so $A$ and $B$ are independent given $D$.

But $A$ and $B$ are independent given $C \union D$ iff $A$ is
independent of $A$, which is false for all nontrivial events $A$.
\end{staffnotes}

\end{solution}

\ppart 
\begin{falseclm*}
If $A$ and $B$ are independent given $C$, and are also independent
given $D$, then $A$ and $B$ are independent given $C \intersect D$.
\end{falseclm*}

\hint Choose $A,B,C,D$ 3-way but not 4-way independent.

\begin{solution}
Choose $A,B,C,D$ 3-way but not 4-way independent.  Then $A$ and $B$
are independent given $C$ or given $D$ by 3-way independence.  But
\begin{align*}
\prcond{A \intersect B}{C \intersect D}
 & \eqdef \frac{\pr{ A \intersect B \intersect C\intersect D}}{\pr{C \intersect D}}\\
 & \neq \frac{\pr{A} \cdot \pr{B} \cdot \pr{C} \cdot \pr{D}}{\pr{C \intersect D}}
      & \text{(not 4-way independent)}\\
 & = \frac{\pr{A} \cdot \pr{B} \cdot \pr{C} \cdot \pr{D}}{\pr{C} \cdot \pr{D}}
      & \text{(2-way independence of $C,D$)}\\
 & = \pr{A} \cdot \pr{B}\\
 & = \prcond{A}{C \intersect D} \cdot \prcond{B}{C \intersect D}
      & \text{(3-way independence)},
\end{align*}
\end{solution}
so $A$ and $B$ are not independent given $C \intersect D$.

\eparts

\end{problem}
