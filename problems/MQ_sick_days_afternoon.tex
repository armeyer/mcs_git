\newcommand{\size}[1]{|#1|}

\begin{problem}
The working days in the next year can be numbered
1, 2, 3, \ldots, 270.  I'd like to avoid as many as possible.
%
\begin{itemize}
\item On even-numbered days, I'll say I'm sick.
\item On days that are a multiple of 3, I'll say I was stuck in
  traffic.
\item On days that are a multiple of 5, I'll refuse to come out from
  under the blankets.
\end{itemize}
%
In total, how many work days will I \emph{avoid} in the coming year?

\begin{solution}
Let $D_2$ be the set of even-numbered days, $D_3$
be the days that are a multiple of 3, and $D_5$ be days that are a
multiple of 5.  The set of days I can avoid is $D_2 \cup D_3 \cup
D_5$.  By the Inclusion-Exclusion Rule, the size of this set is:
%
\begin{align*}
\size{D_2 \cup D_3 \cup D_5}
    & = \size{D_2} + \size{D_3} + \size{D_5} \\
    & \quad - \size{D_2 \cap D_3} - \size{D_2 \cap D_5} - \size{D_3
      \cap D_5}\\
    & \quad + \size{D_2 \cap D_3 \cap D_5} \\
    & = \frac{270}{2} + \frac{270}{3} + \frac{270}{5}
        - \frac{270}{2 \cdot 3} - \frac{270}{2 \cdot 5} - \frac{270}{3
          \cdot 5}
        + \frac{270}{2 \cdot 3 \cdot 5} \\
    & = 198
\end{align*}
\end{solution}

\end{problem}
