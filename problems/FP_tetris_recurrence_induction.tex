\begin{staffnotes}
The motivation for this recurrence is based on a tetris-style tiling
problem~\bref{FP_tetris_recurrence_well_order_variant}.
\end{staffnotes}

\begin{problem}
Define
\[
T_n \eqdef T_{n-1} + 2 T_{n-2} \qquad \text{(for $n \geq 2$)},
\]
where $T_0 = T_1 = 1$.

Prove \textbf{by induction} that
\begin{equation}\tag{*}
T_n  = \frac{2^{n+1}+(-1)^n}{3}
\end{equation}
for all $n \geq 0$.

\begin{solution}
The induction hypotheses $P(n)$ will be equation~(*).

\inductioncase{Base cases}: ($n =0, 1$.)
\begin{align*}
T_0 & = 1 = \frac{2^{(0+1)}+(-1)^0}{3},
T_1 & = 1 = \frac{2^{(1+1)}+(-1)^1}{3}.
\end{align*}
So $P(0)$ and $P(1)$ are true.

\inductioncase{Induction step}.  For $n \geq 2$, we need to prove
$P(n)$ assuming by strong induction that $P(k)$ is true for $0\leq k <
n$.
\begin{align*}
T_n & \eqdef T_{n-1}+2T_{n-2} & \text{(by def since $n\geq 2$)}\\
    & =	\frac{2^{(n-1)+1}+(-1)^{n-1}}{3} + 2\frac{2^{(n-2)+1}+(-1)^{n-2}}{3}
           & (\text{$P(n-1), P(n-2)$ are true by hypothesis})\\
    & = \frac{2^{n}+(-1)^{n-1} + 2^{n} + 2(-1)^{n-2}}{3}\\
    & = \frac{2(2^{n}) - (-1)^{n} + 2 (-1)^n}{3}\\
    & = \frac{2^{b+1}+(-1)^{n}}{3}\\
\end{align*}
This proves $P(n)$, completing the induction step.

It follows by strong induction that~(*) holds for all $n\geq 0$, as
desired.
\end{solution}

\end{problem}
