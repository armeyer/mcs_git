\begin{problem}
What is the coefficient of $x^n$ in the generating function
\begin{eqnarray*}
1 + \frac{x}{(1-x)^2} \quad ?
\end{eqnarray*}
\hfill \examrule[0.7in]
\begin{solution}
\[
\begin{cases}
n & \text{if }n>0,\\
1 & \text{if }n=0.
\end{cases}
\]
\end{solution}

\end{problem}
