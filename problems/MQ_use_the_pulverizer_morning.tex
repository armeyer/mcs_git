\begin{problem}

\mbox{}

\bparts

\ppart Use the \idx{Pulverizer} to find integers $x,y$ such that
\[
x30 + y22 = \gcd(30,22).
\]

\begin{solution}
Let $a = 30, b=22$.  Here is the table produced by the Pulverizer:
\[
\begin{array}{ccccrcl}
x & \quad & y & \quad & \rem{x}{y} & = & x - q \cdot y \\ \hline
 30 &&  22 &&  8 & = &    a - b \\
 22 &&   8 &&  6 & = &    b - 2 \cdot  8 \\
&&&&             & = &    b - 2 \cdot (a - b) \\
&&&&             & = &   -2 \cdot a + 3 \cdot b \\
  8 &&   6 &&  2 & = &    8 - 6  \\
&&&&             & = &   (a - b) 
                         -(-2 \cdot a + 3 \cdot b) \\
&&&&             & = &   3 \cdot a - 4 \cdot b \\
  6 &&   2 &&  0 &   & 
\end{array}
\]
so $(x,y) = (3,-4)$ works.
\end{solution}

\examspace[3.5in]

\ppart Now find integers $x',y'$ with $0 \leq y' < 30$ such that
\[
x'30 + y'22 = \gcd(30,22)
\]
\begin{solution}
Since $(x,y) = (3,-4)$ works, so does $(3 - 22n, -4 + 30n)$ for any $n \in
\integers$, so letting $n=1$, we have
\[
-19 \cdot 30 + 26 \cdot 22 = 2
\]
Another possible answer is $(x',y') = (-8,11)$, since in this case the gcd is 2.
\end{solution}

\eparts
\end{problem}
