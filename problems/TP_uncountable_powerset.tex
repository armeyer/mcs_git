\begin{problem}
Prove that if $S$ is an infinite set, then $\power(S)$ is uncountable.

\begin{solution}
If $S$ is infinite, then so is $\power(S)$, hence if
$\power(S)$ was countable, then by definition
\[
\nngint \bij \power(S).
\]
Also, since $S$ is infinite, \inbook{Lemma~\bref{smallestinf} implies
  that} \inhandout{and we know that $\nngint$ is the ``smallest''
  infinite set, we have}
\[
   S \surj \nngint.
\]
These two relations imply that
\[
S  \surj \power(S),
\]
contradicting Cantor's Theorem\inbook{~\bref{powbig}}.
\end{solution}

\end{problem}
