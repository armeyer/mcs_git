\begin{problem}

Give the DAG with minimum number of edges which has the same positive
walk relation as the given DAG in the figure below.%~\ref{mindag}.

\begin{figure}[h]
\includegraphics[width = 5in]{minimum_DAG}
%\caption{DAG for Problem 2}

\end{solution}


\end{problem}
