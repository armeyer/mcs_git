\begin{problem}

\begin{staffnotes}
As a 2 point problem: 0.5 points for ``1'', 1.5 points for ``$3^n$.''
\end{staffnotes}

\bparts

\ppart

What is the coefficient of $x^n$ in the generating function
$1/(1-x)$?\hfill \examrule[0.7in]

\begin{solution}
1
\end{solution}

\ppart What is the coefficient of $x^n$ in the generating function 
$1/(1-3x)$?\hfill \examrule[0.7in]
\begin{solution}
\[
3^n
\]
\end{solution}
\eparts
