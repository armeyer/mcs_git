\begin{problem}
Indicate the correct choices:

\bparts
\ppart A formula is satisfiable iff its negation is
\begin{itemize}
\item not satisfiable
\item also satisfiable
\item valid
\item not valid
\end{itemize}

\begin{solution}
not valid
\end{solution}

\ppart A formula is valid iff its negation is
\begin{itemize}
\item not valid
\item also valid
\item satisfiable
\item not satisfiable
\end{itemize}

\begin{solution}
not satisfiable
\end{solution}

\ppart Formula $F$ is equivalent to formula $G$ iff
\begin{itemize}
\item $F \QIFF G$ is valid
\item $F \QIFF \QNOT(G)$ is not valid
\item $F \QXOR G$ satisfiable
\item $F \QXOR G$ is not satisfiable
\end{itemize}

\begin{solution}
$F \QIFF G$ is valid.  Also,  $F \QXOR G$ is not satisfiable.
\end{solution}

\eparts
\end{problem}
