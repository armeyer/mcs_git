\begin{problem}

%% type: short-answer
%% title: Relational Properities

For each of the following relations on people, decide whether
the relation is
\begin{enumerate}

\item reflexive

\item irreflexive

\item antisymmetric

\item transitive

\end{enumerate}

%% Answer with a sequence of integers separated by some spaces, for
%% example,
%%             
%% \begin{equation*}
%% 4 2 3
%% \end{equation*}
%% 
%% Don't use commas or periods.

\bparts

\ppart are the same age

\begin{solution}
1, 4
\end{solution}

\ppart is younger than

\begin{solution}
2,
3,
4

actually asymmetric: different people can't be younger than each other
\end{solution}

\ppart have the same parents

\begin{solution}
1,
4
\end{solution}

\ppart is a descendent of

\begin{solution}
2,
3,
4

irreflexive since a person is not considered to be their own
descendent
\end{solution}

\ppart have a parent in common

\begin{solution}
1

different brothers have a parent in commom, so not antisymmetric;

people have children from multiple marriages, so not transitive
\end{solution}

\eparts

\end{problem}
