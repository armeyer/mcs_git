\begin{problem}
\bparts Define the length of a planar embedding $\embed{E}$ of a graph
$G$ to be the sum of the lengths of the faces of $\embed{E}$.  \ppart
Give a formula for $\embed{E}$ in terms of $v$, $c$ and $e$ (number of
$v$ertices, $c$omponents and $e$dges in $G$). \hint{the length of
  face \emph{abca} is 3.}
\examspace[2in]
\begin{solution}
$\embed{E} = 2e.$
\end{solution}
\ppart
Conclude that all embeddings of the same planar graph have the same length.
\examspace
\begin{solution}
Each edge has exactly two occurrences on the faces of an embedding, so
the length of an embedding is always twice the number of edges in the
graph, which is the same for any embedding of the same graph.
\end{solution}

\eparts

\end{problem}
