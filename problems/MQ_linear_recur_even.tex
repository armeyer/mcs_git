\begin{problem}

Let $f_n$ denote the $n^{\textrm{th}}$ non-negative \emph{even} integer,
i.e., $f_0 = 0, f_1 = 2, f_2 = 4,$ and so on.

\bparts

\ppart Give a linear recurrence for $f_n$. 

\begin{solution}
\begin{equation}
f_n = f_{n-1} + 2 .
\label{eq:evenrecur}
\end{equation}
\end{solution}

\examspace[0.75in]

\ppart Express the generating function $F(x)$ for the sequence $f_0,
f_1, \ldots$, i.e,
\[
F(x) ::= f_0 + f_1 x + f_2 x^2 \ldots + f_n x^n + \ldots ,
\]
in closed form as a quotient of polynomials. 

\begin{solution}
We have:
\begin{align*}
F(x) &= f_0 + f_1 x + \ldots + f_n x^n + \ldots , ~~~~~~~~~~~\text{and}\\
-xF(x) &= -f_0 x - f_1 x^2 - \ldots - f_{n-1} x^n + \ldots . 
\end{align*}
Therefore, we get:
\begin{align*}
F(x) (1-x) &= f_0 + (f_1 - f_0) x + (f_2 - f_1) x^2 + \ldots \\
&= 0 + 2x + 2x^2 + \ldots , ~~\text{from the recurrence
  \eqref{eq:evenrecur}} \\
&= \frac{2x}{1-x}.
\end{align*}
So,
\[
F(x) = \frac{2x}{(1-x)^2} .
\]

\end{solution}

\eparts

\end{problem}
