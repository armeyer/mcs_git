\begin{problem}

Let $b$, $c$, $a_{0}$, $a_{1}$, $a_{2}$,\dots be real numbers such
that

\begin{equation*}
a_{n} = b(a_{n-1}) + c
\end{equation*}
for $n \ge 1$.

Let $G(x)$ be the generating function for this sequence.  

\bparts

\iffalse
\ppart
What is the coefficient of $x^{n}$ in the series expansion of $G(x)$?

\begin{solution}
$a_n$
\end{solution}
\fi

\ppart Express the coefficient of $x^{n}$ for $n \ge 1$ in the series
expansion of $bxG(x)$ in terms of $b$ and $a_i$ for suitable $i$.

\begin{solution}
$b(a_{n-1})$
\end{solution}

\ppart
What is the coefficient of $x^{n}$ for $n \ge 1$ in the series expansion of
$cx/(1-x)$?

\begin{solution}
$c$
\end{solution}

\ppart Use the previous results to Exhibit a very simple expression
for $G(x) - bxG(x) - cx/(1-x)$.

\begin{solution}
$a_0$
\end{solution}

\ppart 
Using the method of partial fractions, we can find real numbers $d$
and $e$ such that
\begin{equation*}
G(x) = d/L(x) + e/M(x).
\end{equation*}
What are $L(x)$ and $M(x)$?

\begin{solution}
$(1-x)$ and $(1-bx)$

We have
\[
G(x) - bxG(x) - cx/(1-x) = a_0.
\]
Solving for $G(x)$, we have
\[
    G(x) = cx/(1-x)(1-bx) + a_0/(1-bx).
\]
which by the method of partial fractions  can be expressed in the form
\[
d/(1-x) + e/(1-bx).
\]
\end{solution}

\eparts

\end{problem}
