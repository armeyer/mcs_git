\begin{problem}
\begin{problemparts}

\problempart
Prove that the \idx{average degree} of a tree is less than 2.

\begin{solution}
\begin{proof}
A tree $T$ with $v$ vertices has $v-1$ edges.  By the Handshaking
Lemma,
\[
\sum_{x \in T} \degr{x} = 2(v-1) < 2v.
\]
Dividing both sides by $v$ proves the claim.
\end{proof}
\end{solution}


\problempart
Suppose every vertex in a graph has degree at least $k$.  Explain
why the graph has a path of length $k$.

\hint Consider a longest path.

\begin{solution}
Let $P$ be the longest path in $G$.  Let $x$ be one endpoint of the
path $P$.  Now $x$ is adjacent to at least $k$ other vertices.  All of
these must lie on path $P$; otherwise, the path could be extended.
Therefore, $P$ contains at least $k$ vertices in addition to $x$, and
thus has length at least $k$.
\end{solution}

\end{problemparts}
\end{problem}
