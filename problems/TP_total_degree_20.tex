\begin{problem}
Among \emph{connected} simple graphs whose sum of vertex degrees is
20:

\begin{staffnotes}
a,b worth 5 points each.

part a answer ``10'' gets $-4$.
\end{staffnotes}

\bparts

\ppart What is the largest possible number of vertices?

\exambox{0.5in}{0.5in}{0.0in}

Briefly explain:
\examspace[1.5in]

\begin{solution}
\textbf{11}.

With total degree 20 there are 10 edges.  The largest number of
vertices in a connected graph with 10 edges is a tree with 11 vertices.
\end{solution}

\ppart What is the smallest possible number of vertices?

\exambox{0.5in}{0.5in}{0.0in}

Briefly explain:
\examspace[1.5in]

\begin{solution}
\textbf{5}.

A complete graph has the largest possible number of edges for a given
number of vertices.  The complete graph $K_5$ has 10 edges.
\end{solution}

\eparts

\end{problem}
