\begin{problem}
Prove that if there is a total injective ($[\geq 1 \text{ out}, \leq 1
  \text{ in}]$) relation from $S$ to $\nngint$, then $S$ is
countable.

\begin{solution}
There is a total injective relation $S \to \nngint$ means $S \inj \nngint$.
We know that $S \inj \nngint$ iff $\nngint \surj S$ \inbook{by
  Lemma~\bref{surjinjbij_properties}}, which implies that $S$ is
countable\inbook{ by Lemma~\bref{NsurjC}}.
\end{solution}

\end{problem}
