\begin{problem}
The notation $[x^n]F(x)$ refers to the coefficient of $x^n$ in the
generating function $F(x)$.  \inhandout{Circle}\inbook{Indicate} all
the expressions below that equal $[x^n] 4xG(x)$ (most of them do).
\[\begin{array}{lll}
  4[x^n] xG(x)  \qquad &  4x[x^{n}]G(x) \qquad & [x^{n-1}]4G(x)\\
\\
  \paren{[x^n]4x} \cdot [x^n]G(x) \qquad &  ([x]4x) \cdot [x^n]xG(x) \qquad & [x^{n+1}]4x^2G(x)
\end{array}\]

% [x^n]2xG(x) + 2[x^n]xG(x)

\begin{solution}
\[
  4[x^n] xG(x), \quad [x^{n-1}]4G(x), \quad [x]4x \cdot [x^n]xG(x),
  \quad [x^{n+1}]4x^2G(x)
\]
\end{solution}
\end{problem}
