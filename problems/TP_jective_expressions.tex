\begin{problem}
Let $R:A \to B$ be a binary relation.  Each of the following
formulas expresses the fact that $R$ has a familiar relational
``arrow'' property such as being surjective or being a function.

Identify the relational property expressed by each of the following
relational expressions.  Explain your reasoning.

\bparts

\ppart $R \compose R^{-1} \subseteq \ident{B}$ \hfill \examrule[0.5in]
\begin{solution}
function.
\end{solution}


\ppart $R^{-1} \compose R \subseteq \ident{A}$ \hfill \examrule[0.5in]
\begin{solution}
injection
\end{solution}

\ppart $R^{-1} \compose R \supseteq \ident{A}$ \hfill \examrule[0.5in]
\begin{solution}
total
\end{solution}

\ppart $ R \compose R^{-1} \supseteq \ident{B}$ \hfill \examrule[0.5in]
\begin{solution}
surjection
\end{solution}

\eparts
\end{problem}
