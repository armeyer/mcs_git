\begin{problem}  Prove that
\[  %\begin{equation}\label{v}
[\forall x.\, \QNOT P(x)] \QIMPLIES \QNOT(\exists z.\, P(z)).
\]  %\end{equation}
is valid.  (Use the validity proof from the subsection on Validity in
the Course Notes, namely section 5.1.5, as guides to writing your proof.)

\begin{solution}
\begin{proof}
Assume
\begin{equation}\label{eq:1}
\forall x.\; \QNOT P(x).
\end{equation}

Now we prove by contradiction that $ \QNOT(\exists z.\, P(z))$ holds.
Namely, we assume $\exists z.\, P(z)$ and reach a contradiction.

So suppose $\exists z.\, P(z)$ holds.  So there is an element $c$
such that $P(c)$.  But from~\eqref{eq:1}, we know that $\QNOT(P(c))$
holds, contradicting the fact that $P(c)$ holds.  Hence, the
assumption $\exists z.\, P(z)$ must be false, that is, $\QNOT(\exists
z.\, P(z))$ is true.
\end{proof}
\end{solution}

\end{problem}
