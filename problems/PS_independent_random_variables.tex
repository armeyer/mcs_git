\begin{problem}
  Let $R$ and $S$ be independent random variables, and $f$ and $g$ be any
  functions such that $\domain{f} = \codomain{R}$ and $\domain{g} =
  \codomain{S}$.  Prove that $f(R)$ and $g(S)$ are also independent
  random variables.

\hint The event $[f(R) = a]$ is the disjoint union of all the events
$[R = r]$ for $r$ such that $f(r)=a$.

\begin{solution}
By the hint and the Sum Rule
\[
\pr{f(R) = a} = \sum_{\set{r \suchthat f(r)=a}} \pr{R = r}.
\]
Also, the event $[f(R) = a  \QAND  g(S) = b]$ is the disjoint union
of the events $[R = r \QAND  S = s]$ for pairs $(r,s)$ such that
$f(r)=a$ and $g(s)=b$.  Hence,
\begin{align*}
\lefteqn{\pr{f(R) = a  \QAND  g(S) = b}}\\
& =  \sum_{\set{(r,s) \suchthat f(r)=a\, \QAND\, g(s)=b}}
                \pr{R = r  \QAND  S = s}\\
& =   \sum_{\set{(r,s) \suchthat f(r)=a\, \QAND\, g(s)=b}}
                \pr{R = r}\cdot  \pr{S= s}
            \hspace{0.5in}\mbox{[$R,S$ independent]}\\
& =  \sum_{\set{s \suchthat g(s)=b}}
              \pr{S = s} \paren{\sum_{\set{r \suchthat f(r)=a}}
                   \pr{R = r}}\\
& =  \sum_{\set{s \suchthat g(s)=b}} \pr{S = s} (\pr{f(R) = a})\\
& =  \pr{f(R) = a} \sum_{\set{s \suchthat g(s)=b}} \pr{S = s}\\
& =  \pr{f(R) = a}\cdot \pr{g(S) = b}.
\end{align*}
\end{solution}

\end{problem}
