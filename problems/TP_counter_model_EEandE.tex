\begin{problem}
Find a counter-model showing the following is not valid.
\[
[\exists x.\, P(x)  \QAND\  \exists x . Q(x)]
 \QIMPLIES \exists x. [P(x) \QAND Q(x)]
\]

(Just define your counter-model. You do not need to verify that it is correct.)

\begin{solution}
Let the domain be numbers, and let
\begin{align*}
P(x) & \eqdef [x =0]\\
Q(x) & \eqdef [x =1].
\end{align*}
\end{solution}

\end{problem}
