\begin{problem}

\inhandout{\textbf{Circle}}\inbook{Indicate} which of the following
assertions (there may be more than one) are equivalent to
\[
A \strict \nngint.
\]

\begin{itemize}
\item $\card{A}$ is undefined.

\item $A$ is countably infinite.

\item $A$ is uncountable.

\item $A$ is finite.

\item $\nngint \surj A$.

\item $\forall n \in \nngint$, $\card{A} \leq n$.

\item $\forall n \in \nngint$, $\card{A} \geq n$.

\item $\exists n \in \nngint.\, \card{A} \leq n$.

\item $\exists n \in \nngint.\, \card{A} < n$.

\end{itemize}

\begin{solution}
$A$ is finite;  $\exists n \in \nngint.\, \card{A} \leq n$;
$\exists n \in \nngint.\, \card{A} < n$.
\end{solution}

\end{problem}
