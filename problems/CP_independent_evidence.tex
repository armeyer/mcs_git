\begin{problem}
Event $E$ is \emph{evidence in favor} of event $H$ when $\prcond{H}{E}
> \pr{H}$, and it is \emph{evidence against} $H$ when $\prcond{H}{E} <
\pr{H}$.

\bparts

\ppart Give an example of events $A,B,H$ such that $A$ and $B$ are
independent, both are evidence for $H$, but $A \union B$ is 
evidence against $H$.

\hint Let $\sspace = \Zintv{1}{8}$

\begin{staffnotes}
There may be simpler examples.
\end{staffnotes}

\begin{solution}
\begin{align*}
H & \eqdef \set{3,4,8}\\  %\set{3,4}\\
A & \eqdef \set{1,2,3,4}\\
B & \eqdef \set{3,4,5,6}.
\end{align*}

\end{solution}

\ppart Prove $E$ is evidence in favor of $H$ iff $\setcomp{E}$ is
evidence against $H$.

\begin{solution}
To prove the implication from left to right, suppose to the contrary
that $E$ was evidence for $H$ and $\setcomp{E}$ was not evidence
against $H$.  Then
\begin{align*}
\pr{H} & = \prcond{H}{E}\pr{E} + \prcond{H}{\setcomp{E}}\pr{\setcomp{E}}
             & \text{(Total probability)}\\
       & \geq \prcond{H}{E}\pr{E} + \pr{H}\pr{\setcomp{E}}
              & (\setcomp{E}\text{ is not evidence against } H)\\
       & > \pr{H}\pr{E} + \pr{H}\pr{\setcomp{E}}
              & (E\text{ is evidence in favor of } H)\\
       & = \pr{H}(\pr{E} + \pr{\setcomp{E}})\\
       & = \pr{H},
\end{align*}
which proves $\pr{H} > \pr{H}$, a contradiction.

The implication from right to left now follows by replacing $E$ by
$\setcomp{E}$ in the left to right implication.
\end{solution}

\eparts

\end{problem}
