\newcommand\then\Rightarrow
\renewcommand\iff\Leftrightarrow

\begin{enumerate}
\item
\begin{description}
\item {Infinity:}\hspace{2pc} $\ex S,s\in S \all x\in S\ex y\in S
  (x\in y)$
\item {Induction:} $\neg( \ex S,s\in S \all x\in S\ex y\in S (x\ni y) )$
\end{description}

\item
\begin{description}\item {Extensionality:}
$(\all t (t\in x\iff t\in y))\then (\all t (t\ni x\iff t\ni y)$

\item {Replacement family} (for each ST-defined relation $R_c(x,y)$):\\
$(\all x\ex Y( R_c(x,y)\then y\in Y))\then
(\all X\ex Y(\ex x{\in}X R_c(x,y)\iff y\in Y))$
\end{description}

\end{enumerate}
