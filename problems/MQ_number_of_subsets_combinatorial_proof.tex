\begin{problem}

Find a combinatorial (\emph{not} algebraic) proof that
\[
\sum_{i=0}^n \binom{n}{i} = 2^n.
\]

\examspace[2in]

%\hint Think about the binomial theorem.
\begin{solution}
%Both sides are the sum of the coefficients of $(x + y)^n$.

  There are $2^n$ subsets of $\set{1,2,\dots,n}$.  But the number of
  such subsets is sum of the number of subsets of each possible size.
  Possible sizes are $k \in [0,n]$, and we know there are
  $\binom{n}{k}$ size $k$ subsets.
\end{solution}

\end{problem}
