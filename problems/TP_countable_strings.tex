\begin{problem}
Show that the set $\finbin$ of finite binary strings is countable.

\begin{solution}
List the strings in order of length, with strings of the same length
ordered as binary numbers:

\[
0,1,00,01,10,11,000,001,010,011,100,\dots.
\]

\end{solution}

\end{problem}
