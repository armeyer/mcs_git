\begin{problem}

\iffalse Let $D$ be a digraph.  If $v$ is a vertex of $D$, the
  out-degree $\outdegr{v}$ of $v$ in $D$, is
  \[\mbox{$o(v) ::=$ the number of edges in $D$ directed out of $v$}.\] The
  in-degree of $v$ is
  \[\mbox{$i(v)::=$ the number of edges in $D$ directed into $v$}.\]
  An edge directed from $v$ to itself, $(v,v)$, is counted once for
  $o(v)$ and also once for $i(v)$.\fi

Let $S$ be a nonempty set of size $n \in \integers^+$, and let $f : S
\rightarrow S$ be total function.  Let $D_f$ be the digraph with
vertices $S$ whose edges are $\set{\diredge{s}{f(s)} \suchthat s \in S}$.

\bparts

\ppart What are the possible values of the out-degrees of vertices of
$D_f$?

\begin{solution}
They are all $1$.  The only edge whose tail is vertex $s$ must have
head $f(s)$.  Moreover, since $f$ is total, of every vertex $s$ there
is an edge whose tail is $s$.
\end{solution}

\ppart What are the possible values of the in-degrees of the vertices?

\begin{solution}
They can be any sequence of $n$ nonnegative integers whose sum, by the
Handshaking Lemma\inbook{~\bref{digraph-handshake}}, is $n$.
\end{solution}

\ppart Suppose $f$ is a surjection.  Now what are the possible values
of the in-degrees of the vertices?

\begin{solution}
Since $S$ is finite and $f$ is total, $f$ is a surjection iff $f$ is a
bijection.  So there is a unique edge into each vertex, that is, all
in-degrees are equal to $1$.
\end{solution}

\eparts

\end{problem}
