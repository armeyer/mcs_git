\begin{problem}
\bparts

\ppart List the set $\fmv{4}$ of possible Nim games that can result
from the making the first move in the Nim game \nimg{4}\ with one
size-four pile.

\begin{solution}
\nimg{\emptyset}, \nimg{1}, \nimg{2}, \nimg{3}.
\end{solution}

\ppart What is the representation of \nimg{4} as a pure set ``built out of nothing?''

\begin{solution}
\begin{align*}
\left\{ \right. & \\
& \emptyset,\\
& \set{\emptyset},\\
& \set{\emptyset,\ \ \set{\emptyset}},\\
& \set{\emptyset,\ \ \set{\emptyset},\ \ \set{\emptyset,\ \ \set{\emptyset}}}\\
& \left. \right\}
\end{align*}
\end{solution}

\ppart  What piles of stones are there is the Nim game whose set representation is
\[
\set{\set{\set{\set{\emptyset}}}}
\]

\begin{solution}
This is $\nimg{1,1,1,1}$ with four size-one piles.
\end{solution}

\eparts
\end{problem}
