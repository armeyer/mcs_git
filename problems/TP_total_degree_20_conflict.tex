\begin{problem}
How many vertices could each of the following graphs have, at most?

\bparts

\ppart A connected simple graph with total degree (sum of the degrees
of all vertices) equal to $20$.

\exambox{0.5in}{0.5in}{0.0in}

\examspace[1.5in]
\begin{solution}
\textbf{11}.

With total degree 20 there are 10 edges.  The largest number of
vertices in a connected graph with 10 edges is a tree with 11 vertices.
\end{solution}

\ppart A complete graph where each vertex has degree $4$.

\exambox{0.5in}{0.5in}{0.0in}

\examspace[1.5in]

\begin{solution}
\textbf{5}.
A complete graph of degree $4$ must have $5$ vertices.
\end{solution}

\end{problem}
