\begin{problem} \mbox{}

 Define a function $f(n)$ such that $f = \Theta(n^2)$
  and $\QNOT (f \sim n^2)$.

\begin{solution}
Let $f(n) \eqdef 2n^2$.
\end{solution}

  \examspace[0.3in]
  \[
  f(n) = \inhandout{\instatements{\fbox{\rule{4.0in}{0in}\rule[-0.35]{0in}{0.75in}}}}%
         \insolutions{2n^2}
  \]

  \examspace[0.3in]

\end{problem}
