\begin{problem}
Circle each of the true statements below. 

Explanations are not required, but partial credit for wrong answers will not be given without them. 

\begin{itemize}

\item $\displaystyle n = \Theta \left(\frac{3n^3}{(n+1)(n-1)}\right)$

\item $\displaystyle n^3 \sim n^{3} + n^{2}$

\item $\displaystyle 3^n=O\left(2^n\right)$

\item $\displaystyle \ln \left(\left(n^2\right)!\right) = O\left(n^2\ln(n)\right)$ 

%\item $\displaystyle n^{\sin (n\pi/2) + 1} = o\left(n^2\right)$

\end{itemize}

\begin{solution}The 1st, 2nd, and 4th statements are true.

\begin{itemize}

\item $3n^3 = \Theta \left(n^3\right)$. $(n+1)(n-1) = \Theta \left(n^2\right)$. 
Therefore $\frac{3n^3}{(n+1)(n-1)} = \Theta \left(n\right)$.

\item $\frac{n^{3} + n^{2}}{n^3} = \frac{n^3}{n^3} + \frac{n^2}{n^3}$. The ratio approaches $1 + 0 = 1$, 
therefore the two expressions are similar.

\item $\lim_{n\to\infty}{\frac{3^n}{2^n}} = \lim_{n\to\infty}{(3/2)^n} = \infty$

\item $\ln \left(\left(n^2\right)!\right) = O\left(n^{2}\ln(n^{2})\right) = O\left(n^2\ln(n)\right)$ 
with some abuse of notation in the second part. The first part is due to Stirling's approximation, and the second part 
is due to turning the exponent inside the log into a constant factor of 2 outside which does not matter asymptotically.

\end{itemize}

\end{solution}

\end{problem}
