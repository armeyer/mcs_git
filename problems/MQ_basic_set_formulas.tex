\begin{problem}
\inhandout{A \emph{formula of \idx{set theory}} is a predicate formula
  that only uses the predicate ``$x \in y$.''  The domain of discourse
  is the collection of sets, and ``$x \in y$'' is interpreted to mean
  the set $x$ is one of the elements in the set $y$.

For example, since $x$ and $y$ are the same set iff they have the same members, here's how we
can express $x=y$ with a formula of set theory:
\[
\forall z.\, (z \in x\ \QIFF\ z \in y).
\]}

Express the following assertions about sets by a formula of set theory.

\bparts

\ppart $x = \emptyset$.

\begin{solution}
$\forall z.\, \QNOT(z\in x)$.

\begin{staffnotes}
1 pt.
\end{staffnotes}

\end{solution}

\examspace[1in]

\ppart $x = \set{y,z}$.

\begin{solution}
$\forall w.\, w \in x \QIFF (w=y \QOR\ w = z)$.

\begin{staffnotes}
2 pts.
\end{staffnotes}
\end{solution}

\eparts

\end{problem}
