\begin{problem}
Suppose let $f$ and $g$ be real-valued functions.

\bparts

\ppart Give an example of $f,g$ such that
\[
\limsup_{} fg < \limsup_{} f \cdot \limsup_{} g,
\]
and all the $\limsup_{}$'s are finite.

\begin{solution}
Let $f(x) \eqdef \sin x$ and $g(x) \eqdef \cos x$.

Then $\limsup_{} f = 1 = \limsup_{} g$ but $\limsup_{} fg = \sqrt{2}/2$.

Note that using $1/2+ \sin(x)^2$ and $1/2+ \cos(x)^2$ is a similar example with both functions
positive.

\end{solution}

\ppart Give an example of $f,g$ such that
\[
\limsup_{} fg > \limsup_{} f \cdot \limsup_{} g.
\]
and all the $\limsup_{}$'s are finite.

\begin{solution}
Let $f(x) \eqdef \sin x - 1/2 = g(x)$.

Then $\limsup_{} f = 1/2$ and
\[
\paren{\limsup_{} f}^2 = \paren{\frac{1}{2}}^2 \neq
\paren{-\frac{3}{2}}^2 = \limsup_{} f^2.
\]
\end{solution}

\begin{staffnotes}
If $f,g \geq 0$ then 
\[
\limsup_{} fg \leq \limsup_{} f \cdot \limsup_{} g
\]
and
\[
\limsup_{} f^2 = \paren{\limsup_{} f}^2.
\]

\end{staffnotes}

\eparts

\end{problem}
