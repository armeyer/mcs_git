\begin{problem}
Give a combinatorial proof for this identity:
\[
\sum_{\substack{i+j+k=n\\ i,j,k \ge 0}} \binom{n}{i,j,k} = 3^n
\]

\begin{solution}
Number of sequences of length $n$ composed of digits $0$,$1$, and $2$.
\end{solution}

\end{problem}
