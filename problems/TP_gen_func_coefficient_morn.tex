\begin{problem}
What is the coefficient of $x^n$ in the generating function
\begin{eqnarray*}
\frac{1+x}{(1-x)^2}\quad?
\end{eqnarray*}
\hfill \examrule[0.7in]
\begin{solution}
\[
(n+1) + n = 2n+1
\]
\end{solution}

\end{problem}
