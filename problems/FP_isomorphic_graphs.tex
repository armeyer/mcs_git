\begin{problem}
For each of the following pairs of simple graphs, either define an
isomorphism between them, or prove that there is none.  (We write $ab$
as shorthand for $\edge{a}{b}$.)

\bparts

\ppart
\begin{align*}
G_1\text{ with }& V_1 = \set{1,2,3,4,5,6},\ E_1= \set{12, 16, 23, 25, 34, 45, 56} \\
G_2\text{ with }& V_2 = \set{a,b,c,d,e,f},\ E_2= \set{ab, ad, bc, be, cd, cf, ef} \\
\end{align*}

\examspace[2in]


\begin{solution}
Isomorphic.  There are four isomorphisms possible:
\[
123456 \to  \begin{cases}
              abefcd\\
              dcfeba\\
              ebadcf\\
              fcdabe
              \end{cases}
\]
\end{solution}

\ppart
\begin{align*}
G_1\text{ with }& V_1 = \set{1,2,3,4,5,6},\ E_1=  \set{12, 13, 16, 23, 34, 35, 45, 46} \\
G_2\text{ with }& V_2 = \set{a,b,c,d,e,f},\ E_2=  \set{ab, af, bc, bf, cd, ce, de, ef} \\
\end{align*}

\begin{solution}
Non-isomorphic, vertex $3$ in $G_1$ has degree 4 while all vertices of
$G_2$ have degree at most 3.
\end{solution}

\eparts
\end{problem}
