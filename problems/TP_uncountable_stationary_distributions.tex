\begin{problem}
Explain why there are an uncountable number of stationary
distributions for the following random walk graph.% in Figure~\ref{fig:sinky}.

%\begin{figure}[h]
\begin{center}
\graphic[height=1in]{randomWalkFigs/sinky}
\end{center}

%\caption{}
%\label{fig:sinky}
%\end{figure}

\begin{solution}
Any distribution with $d(b)=d(c)=0$, and $d(a) = p$ and $d(d) = 1-p$
is stationary for all real numbers $p$ in the unit interval.  There
are uncountably many such $p$ (see Problem~\bref{PS_unit_interval}).
\end{solution}

\end{problem}
