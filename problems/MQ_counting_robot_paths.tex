\begin{problem}
There is a robot that steps between integer positions in 3-dimensional
space.  Each step of the robot increments one coordinate and leaves the
other two unchanged.

\bparts

\ppart How many paths can the robot follow going from the origin
$(0,0,0)$ to $(3,4,5)$?

\begin{solution}
\[
\binom{12}{3,4,5}
\]
\end{solution}

\ppart How many paths can the robot follow going from the origin
$(i,j,k)$ to $(m,n,p)$?

\begin{solution}
\[
\binom{m+n+p-(i+j+k)}{m-i,n-j,p-k}
\]

\end{solution}

\eparts

\end{problem}
