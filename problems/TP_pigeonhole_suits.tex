\begin{problem}
What is smallest number $n$ such that set of $n$ cards must have $6$
cards of the same suit?\hfill \examrule[0.7in]

\begin{staffnotes}
No part credit.
\end{staffnotes}

\begin{solution}
\[
21
\]
This is the smallest $n$ such that $\ceil{n/4} = 6$.
\end{solution}

\end{problem}
