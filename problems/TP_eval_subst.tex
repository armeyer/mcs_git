\begin{problem}
\bparts

\ppart\label{TP_es3x}
Write out the evaluation of
\[
\meval{\msubst{3x}{x(x-1)}}{2}
\]
according to the Environment Model and the Substitution Model,
indicating where the rule for each case of the recursive definitions
of $\meval{}{}$ and $\subst{}{}$ or substitution is first used.
Compare the number of arithmetic operations and variable lookups.

\ppart Describe an example along the lines of part~\eqref{TP_es3x}
where the Environment Model would perform 6 fewer multiplications than
the Substitution model.  You need \emph{not} carry out the evaluations.

\begin{solution}

$\meval{\msubst{3x}{x^7}{2}}$

\end{solution}

\ppart Describe an example along the lines of part~\eqref{TP_es3x}
where the Substitution Model would perform 6 fewer multiplications
than the Environment model.  You need \emph{not} carry out the
evaluations.

\begin{solution}

$\meval{\msubst{x^7}{\mtt{5}}{2}}$

\end{solution}

\eparts
\end{problem}
