\begin{problem}
Peeta bakes between $1$ and $2n$ loaves of bread to sell every day.
Each day he rolls a fair, $n$-sided die to get a number from 1 to $n$,
then flips a fair coin.  If the coin is heads, he bakes $m$ loaves of
bread , where $m$ is the number on the die that day, and if the coin
is tails, he bakes $2m$ loaves.

\begin{problemparts}

\ppart For any positive integer $k \le 2n$, what is the probability
that Peeta will make $k$ loaves of bread on any given day?

\hint Express your solution by cases.

\begin{solution}
Each possible outcome on any given day has the form of $(c,d)$, where
$c \in \set{H,T}$ and $d \in \Zintv{1}{n}$.  If $c=H$, every $k<n$ is
possible to be the number of loaves of bread, and if $c=T$, every even
$k$ is possible to be the number.  Therefore, there are 4 possible
cases:

\begin{enumerate}[i]

\item If $k$ is odd and $k \le n$, then the probability is $\frac{1}{2n}$.

\item If $k$ is even and $k \le n$, then the probability is $\frac{1}{n}$.

\item If $k$ is odd and $k > n$, then the probability is $0$.

\item If $k$ is even and $k > n$, then the probability is $\frac{1}{2n}$.

%\end{itemize}

\end{enumerate}

\end{solution}

\ppart What is the expected number of loaves that Peeta would bake on any
given day?

\begin{solution}
Since each possible dice-coin outcome is equally likely, we can sum
the number of loaves for each outcome and then divide the sum by the
total number of outcomes, that is, multiply the sum by the probability
of individual outcome.

\begin{itemize}

\item If $c=H$, then the sum of the $n$ possible outcomes is
$\sum_{i=1}^{n}{i} = \frac{n(n+1)}{2}$.

\item If $c=T$, then the sum of the $n$ possible outcomes is doubled:
$\sum_{i=1}^{n}{2i} = n(n+1)$.

\end{itemize}

Therefore, the expected value is 
\[
\frac{1}{2n}\left(\frac{n(n+1)}{2} + n(n+1)\right) = \frac{3}{4}(n+1).
\]

\end{solution}

\ppart Continuing this process, Peeta bakes bread every day for 30 days.
What is the expected total number of loaves that Peeta would bake?
\begin{solution}
\[
30 \paren{\frac{3}{4}(n+1)} = \frac{45}{2}(n+1)
\]
\end{solution}

\end{problemparts}

\end{problem}
