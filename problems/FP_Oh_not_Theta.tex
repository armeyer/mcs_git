\begin{problem} \mbox{}

 \bparts

 \ppart Define a function $f:\nngint\to\reals$ such that $f(n) = \Theta(n^2)$
  and $\QNOT (f(n) \sim n^2)$. Explain why your function satisfies these conditions.

\begin{solution}
Let $f(n) \eqdef 2n^2$. This is $\Theta(n^2)$ because $\lim_{n\to\infty}2n^2 / n^2 = 2$, and $f(n)\nsim n^2$ because the previous limit does not equal $1$.
\end{solution}

  \examspace[0.1in]
  \[
  f(n) = \examboxplain{4.0in}{0.75in}{e-0.35}\insolutions{2n^2.}
  \]

  \examspace[1.5in]

 \ppart Define a function $g:\nngint\to\reals$ such that $g(n) = O(n^2)$,
  $g(n) \neq \Theta(n^2)$, $g(n) \neq o(n^2)$, and $n = O(g(n))$, and explain why your example works.

\begin{solution}
Let $g(n) \eqdef \paren{n\sin(n\pi/2)}^2 + n$.

For odd $n$, we have $g(n) = n^2 + n$, which implies that $g \neq
o(n^2)$.  For even $n$, we have $g(n) = n$, which implies $n^2 \neq O(g)$
and hence $g \neq \Theta(n^2)$.

\end{solution}

  \examspace[0.1in]
  \[
  g(n) = \examboxplain{4.0in}{0.75in}{e-0.35}%
         \insolutions{n^2\sin(\frac{n\pi}{2}) + n\cos(\frac{n\pi}{2}).}
  \]

 \eparts
\end{problem}
