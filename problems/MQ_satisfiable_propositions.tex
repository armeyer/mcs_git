\begin{problem}
For the following propositions, indicate whether they are valid
(\textbf{V}), satisfiable but not valid (\textbf{S}), or not
satisfiable (\textbf{N}).  For propositions marked \textbf{S}, give an
assignment of truth values that makes the proposition \True\ and an
assignment that makes it \False.
 
\bparts

 \ppart $P \QIMP \bar{P}$
\examspace[0.4in]

 \ppart $P \QAND Q \QAND Z \QAND (P \QIMP \bar{Q})$
\examspace[0.4in]

 \ppart ($P \QAND Q \QAND (\bar{P} \QOR \bar{Q})) \QIMP [((Z \QAND (P
   \QIMP Q)) \QOR (P \QAND \bar{P}))]$
\examspace[0.4in]

 \ppart $(P \QAND Q) \QOR (P \QIMP Q)$
\examspace[0.4in]

 \ppart $(P \QOR \bar{P} \QOR Q) \QIMP (Q \QAND (Q \QIMP \bar{Q}))$
\examspace[0.4in]

\begin{solution}
\begin{itemize}

\item S. $P: \false$; $P: \true$.

\item N, since $P \QAND (P \QIMP \bar{Q})$ means $Q$ must be False, but
  we also know it must be True.

\item V, since the left-hand side is always False

\item S. $P,Q:\true$ or $P:\false,Q:\text{[don't care]}$; $Q:\false, P:\true$

\item N, since the left hand side is always \True (because of $P \QOR
  \bar{P}$) but the right hand side must be \False.
\end{itemize}

\end{solution}

\eparts

\end{problem}
