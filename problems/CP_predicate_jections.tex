\begin{problem}

Five basic properties of binary relations $R:A \to B$ are:

\begin{enumerate}

\item\label{assert:surj} $R$ is a surjection $[\geq 1\ \text{in}]$

\item\label{assert:inj} $R$ is an injection $[\leq 1\ \text{in}]$

\item\label{assert:func} $R$ is a function $[\leq 1\ \text{out}]$

\item\label{assert:total} $R$ is total $[\geq 1\ \text{out}]$

\item\label{assert:empty} $R$ is empty $[= 0\ \text{out}]$

%\item $R$ a bijection$[= 1\ \text{in,out}]$\hfill\examrule[1in]

\end{enumerate}

Below are some assertions about $R$.  For each assertion,
\inbook{indicate}
\inhandout{write the numbers of} all the properties above that the relation $R$
must have\inhandout{; write ``none'' if $R$ might not have any of these
properties}.  For example, \inhandout{you should write ``\eqref{assert:surj},
\eqref{assert:total}'' next to the first assertion.} \inbook{the first assertion impllies that $R$ 
is a total surjection.}
Variables $a,a_1,\dots$ range over $A$ and $b,b_1,\dots$ range over
$B$.

\bparts

\ppart $\forall a\, \forall b.\ a \mrel{R} b$.\inhandout{\hfill\examrule[0.2in]\underline{\eqref{assert:surj}, \eqref{assert:total}}\examrule[0.4in]}

\inbook{\begin{solution}
total \eqref{assert:total}, surjection \eqref{assert:surj}, as stated above.
\end{solution}
}

\ppart $\QNOT(\forall a\, \forall b.\ a \mrel{R} b)$.\hfill\examrule[1.0in]
\begin{solution}
none
\end{solution}

\ppart $\forall a\, \forall b.\ \QNOT(a \mrel{R} b)$.\hfill\examrule[1.0in]
\begin{solution}
empty \eqref{assert:empty}
\end{solution}

\ppart $\forall a\, \exists b.\ a \mrel{R} b$.\hfill\examrule[1in]
\begin{solution}
total \eqref{assert:total}
\end{solution}

\ppart $\forall b\, \exists a.\ a \mrel{R} b$.\hfill\examrule[1in]
\begin{solution}
surjection \eqref{assert:surj}
\end{solution}

\ppart $R$ is a bijection.\hfill\examrule[1in]
\begin{solution}
all the properties \eqref{assert:surj}, \eqref{assert:inj}, \eqref{assert:func}, \eqref{assert:total}
\end{solution}

\ppart $\forall a\, \exists b_1\, a \mrel{R} b_1 \AND \forall b.\, a
\mrel{R} b \QIMPLIES b = b_1$.\hfill\examrule[1in]
\begin{solution}
total function  \eqref{assert:func}, \eqref{assert:total}
\end{solution}

\ppart $\forall a,b.\ a \mrel{R} b \QOR a \neq b$.\hfill\examrule[1in]
\begin{solution}
none
\end{solution}

\ppart $\forall b_1,b_2,a.\ (a \mrel{R} b_1 \QAND a \mrel{R} b_2) \QIMPLIES b_1 = b_2$.\hfill\examrule[1in]
\begin{solution}
function \eqref{assert:func}
\end{solution}

\ppart $\forall a_1,a_2,b.\ (a_1 \mrel{R} b \QAND a_2 \mrel{R} b) \QIMPLIES a_1 = a_2$. \hfill\examrule[1in]
\begin{solution}
injection \eqref{assert:inj}
\end{solution}

\ppart $\forall a_1,a_2,b_1,b_2.\ (a_1 \mrel{R} b_1 \QAND a_2 \mrel{R}
  b_2 \QAND a_1 \neq a_2) \QIMPLIES b_1 \neq b_2$. \hfill\examrule[1in]
\begin{solution}
injection \eqref{assert:inj}
\end{solution}

\ppart $\forall a_1,a_2,b_1,b_2.\ (a_1\mrel{R}b_1 \QAND a_2\mrel{R}b_2
  \QAND b_1\neq b_2) \QIMPLIES a_1\neq a_2$.\hfill\examrule[1in]
\begin{solution}
function \eqref{assert:func}
\end{solution}

\iffalse
\ppart Give an example of a relation $R$ that is a total injective
function from a set $A$ to itself that is not a surjection.

\begin{solution}
Let $A = \nngint$ and $R$ be the successor function $R(n) = n+1$
(more precisely, $R(n) = \set{n+1}$).  There are many other possible
examples, but $A$ must be infinite.
\end{solution}
\fi

\eparts

\end{problem}
