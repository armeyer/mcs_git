\begin{problem}
Outside of their hum-drum duties as Math for Computer Science Teaching
Assistants, Oscar is trying to learn to levitate using only intense
concentration and Liz is trying to become the world champion flaming
torch juggler.  Suppose that Oscar's probability of success is $1/6$,
Liz's chance of success is $1/4$, and these two events are
independent.

\bparts

\ppart If at least one of them succeeds, what is the probability that
Oscar learns to levitate?

\begin{solution}Let $L$ be the event that Oscar learns to levitate, and let
$F$ be the event that Liz becomes the flaming torch juggler champion.
  We can work out the desired probability as follows:

\begin{align*}
\prcond{L}{(L \union F)}
	& = \frac{\pr{L \intersect (L \union F)}}{\pr{L \union F}} \\
	& = \frac{\pr{L}}{1 - \pr{\setcomp{L} \intersect \setcomp{F}}} \\
	& = \frac{1/6}{1 - (1 - 1/6)(1 - 1/4)} \\
	& = \frac{4}{9}
\end{align*}

The first step uses the definition of conditional probability.  In the
second step, we rewrite both the top and bottom of the fraction using
set identities.  Then we substitute in the given probability and
simplify.
\end{solution}

\ppart If at most one of them succeeds, what is the probability that
Liz becomes the world flaming torch juggler champion?

\begin{solution}
Define events $L$ and $F$ as before.

\begin{align*}
\prcond{F}{(\setcomp{L} \union \setcomp{F})}
	& = 	\frac{\pr{F \intersect (\setcomp{L} \union \setcomp{F})}}
	       	{\pr{\setcomp{L} \union \setcomp{F}}} \\
	& = 	\frac{\pr{F \intersect \setcomp{L}}}
	       	{1 - \pr{L \intersect F}} \\
	& = 	\frac{(1/4) \cdot (5/6)}{1 - (1/6) \cdot (1/4)} \\
	& = 	\frac{5}{23}
\end{align*}
\end{solution}

\ppart If exactly one of them succeeds, what is the probability that
it is Oscar?

\begin{solution}
\begin{align*}
\prcond{L}{((L \intersect \setcomp{F}) \union (\setcomp{L} \intersect F))}
	& = \frac{\pr{L \intersect \setcomp{F}}}
			{\pr{((L \intersect \setcomp{F}) \union
			(\setcomp{L} \intersect F))}} \\
	& = 	\frac{(1/6) \cdot (3/4)}
			{(1/6) \cdot (3/4) + (5/6) \cdot (1/4)} \\
	& = 	\frac{3}{8}
\end{align*}
\end{solution}

\eparts
\end{problem}
