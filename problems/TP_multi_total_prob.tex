\begin{problem}
State and prove a version of the Law of Total Probability that applies
to disjoint events $E_1, \dots, E_n$ whose union is the whole sample
space.

\begin{solution}
\begin{rul*}[Law of Total Probability: $n$ Events]
\[
\pr{A} = \sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i},
\]
\end{rul*}

\begin{proof}
Since the $E_i$'s are disjoint, so are the events $A \intersect E_i$,
and since the sample space $sspace = \lgunion_{i=1}^n E_i$, we have
\begin{align}
A & = A \intersect \sspace\notag\\
  & = A \intersect \lgunion_{i=1}^n E_i\notag\\
  & = \lgunion_{i=1}^n (A \intersect E_i)
      & \text{(distributivity of $\intersect$ over $\union$)}\label{1nAiEi}.
\end{align}
So applying the Sum Rule to~\eqref{1nAiEi},
\begin{align*}
\pr{A} & = \sum_{i=1}^n \pr{A \intersect E_i}
            & \text{(Sum Rule)}\\
       & = \sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}
           & \text{(def of conditional probability)}.
\end{align*}

\end{proof}
\end{solution}
\end{problem}
