\begin{problem}

\bparts

\ppart How many digraphs are there with vertex set $\set{1,2,\dots,n}$?\hfill \examrule[0.7in]

Briefly explain.

\examspace[2in]

\begin{solution}
There are $n^2$ potential edges, each of which may or
may not appear in a given graph.  Therefore, the number of graphs is:
\[
2^{n^2}
\]
\end{solution}

\ppart How many asymmetric binary relations on the set
$\set{1,2,\dots,n}$ are there?\hfill \examrule[0.7in]

Briefly explain.

\begin{solution}
There are no self-loops in an asymmetric relation and for each of the
$\binom{n}{2}$ sets of two elements $a \neq b$, either
\begin{enumerate}
\item $a \mrel{R} b$, or
\item $b \mrel{R} a$, or
\item neither,
\end{enumerate}
but not both.  Therefore, the number of asymmetric binary relations is
\[
3^{\binom{n}{2}}.
\]
\end{solution}

\eparts
\end{problem}
