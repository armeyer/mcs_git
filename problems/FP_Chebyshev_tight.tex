\begin{problem}
\iffalse
For any random variable $R$ with mean $\mu$ and standard deviation
$\sigma$, Chebyshev's Theorem states that for any real number $x > 0$,
\[
\pr{\abs{R-\mu} \geq x} \leq \frac{\variance{R}}{x^2}
\]
\fi For any positive real numbers $x$ and $v$, there exists an $R$
with expectation $0$ and variance $v$ for which the Chebyshev Bound
is tight---that is,
\begin{equation}\label{tightcheb}
\pr{\abs{R} \geq x} = \frac{v}{x^2}.
\end{equation}
Describe a PDF---in terms of $x$ and $v$---of such an $R$, and
verify that equation~\eqref{tightcheb} is satisifed.

\iffalse
\hint Let $R$ only take values $0, -x,$ and $x$.
\fi
\begin{solution}

We seek an $R$ with $\expect{R}=\mu=0$ and $\variance{R} = v$ 
that satisfies equation~\eqref{tightcheb}.

Let $\range{R}=\set{-x,0,x}$.  Then,
\[
0 = \expect{R} = x\pr{R=x} - x\pr{R=-x} = x\paren{\pr{R=x} - \pr{R=-x}}.
\]
This implies, since $x>0$,
\[
\pr{R=x} = \pr{R=-x}.
\]
Now,
\[
\variance{R} = \expect{R^2} - \paren{\expect{R}}^2 =
\expect{R^2} = x^2\pr{R=-x} + x^2\pr{R=x} = 2x^2\pr{R=x}.
\]
Hence,
\[
\pr{R=x} = \frac{v}{2x^2}.
\]
This implies
\[
\pr{R=0} = 1 - 2\pr{R=x} = 1 - \frac{v}{x^2},
\]
This completes the description of the distribution of $R$.  Witness that
\[
\pr{\abs{R} \geq x} = \pr{R=-x}+\pr{R=x} = 2\pr{R=x} = \frac{v}{x^2}.
\]
Therefore $R$ satisfies~\eqref{tightcheb}, as required.
\end{solution}

\end{problem}
