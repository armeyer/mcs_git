\begin{problem}
Let
\[
S \eqdef \sum_{n=1}^5 \sqrt{3}{n}\, .  %  ^{\frac{1}{3}}
\]
Using the \idx{Integral Method}\inbook{ of
  Section~\bref{sec:integral_method}}, we can find integers $a$, $b$,
$c$, $d$ and a real number $e$ such that
\begin{equation*}
    \int_a^b x^e \,dx \le S \le \int_c^d x^e \,dx
\end{equation*}

What are appropriate values for $a, \dots, e$?

\begin{solution}
0,

5 (anything),

1,

6 (anything),

1/3, respectively.
\end{solution}

\end{problem}
