\begin{problem}
Show that
\[
\ln \paren{\paren{n^2}!} = O\paren{n^2\ln(n)}.
\]

\examspace[3in]

\begin{solution}
\begin{align*}
\ln \paren{\paren{n^2}!}
   & = \ln \paren{\prod_{i=1}^{n^2} i}\\
   & = \sum_{i=1}^{n^2} \ln i\\
   & \leq \sum_{i=1}^{n^2} \ln n^2\\
   & = n^2\cdot 2 \ln n\\
   & = O(n^2\ln n).
\end{align*}
\end{solution}


\end{problem}
