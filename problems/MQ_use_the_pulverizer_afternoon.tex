\begin{problem}

\mbox{}

\bparts

\ppart Use the \idx{Pulverizer} to find integers $x,y$ such that
\[
x38 + y28 = \gcd(38,28).
\]

\begin{solution}
Here is the table produced by the Pulverizer:
\[
\begin{array}{ccccrcl}
x & \quad & y & \quad & \rem{x}{y} & = & x - q \cdot y \\ \hline
 38 &&  28 &&  10 & = &    38 - 28 \\
 28 &&  10 &&  8 & = &    28 - 2 \cdot  10 \\
&&&&             & = &    28 - 2 \cdot (38 - 28) \\
&&&&             & = &   -2 \cdot 38 + 3 \cdot 28 \\
  10 &&   8 &&  2 & = &    10 - 8  \\
&&&&             & = &   (38 - 28) 
                         -(-2 \cdot 38 + 3 \cdot 28) \\
&&&&             & = &   3 \cdot 38 - 4 \cdot 28 \\
  8 &&   2 &&  0 &   & 
\end{array}
\]

so $(x,y) = (3,-4)$ works.
\end{solution}

\examspace[5in]

\ppart Now find integers $x',y'$ with $0 \leq y' < 38$ such that
\[
x'38 + y'28 = \gcd(38,28)
\]
\begin{solution}
Since $(x,y) = (3,-4)$ works, so does $(3 - 28n, -4 + 38n)$ for any $n \in
\integers$, so letting $n=1$, we have
\[
-25 \cdot 38 + 34 \cdot 28 = 2
\]
Another possible answer is $(x',y') = (-11,15)$, since in this case the gcd is 2.
\end{solution}

\eparts
\end{problem}
