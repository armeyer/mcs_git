\begin{problem}
For any random variable $R$ with mean $\mu$ and standard deviation
$\sigma$ the Chebyshev bound says that for any real number $x > 0$,
\[
\pr{\abs{R-\mu} \geq x} \leq \paren{\frac{\sigma}{x}}^2.
\]
Show that for any real number $\mu$ and real numbers $x \geq \sigma >0$,
there is an $R$ for which the Chebyshev bound is tight, that is,
\begin{equation}\label{cheb=}
\pr{\abs{R-\mu} \geq x} = \paren{\frac{\sigma}{x}}^2.
\end{equation}

\hint First assume $\mu = 0$ and let $R$ take only the values $0, -x$ and
$x$.

\begin{solution}
With the hint, we aim to find an $R$ with $\expect{R}=0$ and $\variance{R}
= \sigma^2$ that satisfies equation~\eqref{cheb=}.

Using the additional hint that $R$ takes only the values $0,-x,x$, we have
\[
0 = \expect{R} = x\pr{R=x} - x\pr{R=-x} = x\paren{\pr{R=x} - \pr{R=-x}},
\]
so
\begin{equation}\label{x-x}
\pr{R=x} = \pr{R=-x},
\end{equation}
since $x>0$.  Also,
\[
\sigma^2  = \expect{R^2}  = x^2\pr{R=-x} + x^2\pr{R=x}  = 2x^2\pr{R=x},
\]
so
\[
\pr{R=x} = \frac{\sigma^2}{2x^2}.
\]
This implies
\[
\pr{R=0} = 1 - 2\pr{R=x} = 1- \paren{\frac{\sigma}{x}}^2,
\]
which completely determines the distribution of $R$.  Moreover,
\[
\pr{\abs{R} \geq x} = \pr{R=-x}+\pr{R=x} = 2\pr{R=x} = \paren{\frac{\sigma}{x}}^2,
\]
which confirms~\eqref{cheb=}.

Finally, given $\mu,x$ and $\sigma$, define $R' \eqdef R+\mu$.  Since
$\variance{R'} = \variance{R}$, the random variable $R'$ will have
mean $\mu$ and standard deviation $\sigma$, and~\eqref{cheb=} will hold
with $R'$ in place of $R$.
\end{solution}

\end{problem}
