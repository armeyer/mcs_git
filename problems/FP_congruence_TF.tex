\begin{problem} \mbox{}

Indicate which of the following statements about \textbf{congruence
  modulo} $n$, where $n > 1$, are \True\ or \False, and
\emph{provide counterexamples} for those that are \False.

\bparts

\iffalse
\ppart If $a c \equiv b c \pmod{n}$ and $n$ does not divide $c$, then $a \equiv b \pmod{n}$.
\hfill \True\ \qquad \False\ \examspace[0.4in]

\begin{solution}
\False.  Need $c$ relatively prime to $n$.  Counterexample:
$n=2 \cdot 3, a=0, b=2, c=3$
\end{solution}
\fi

\ppart If $a \equiv b \pmod{\phi(n)}$ for $a, b > 0$, then $c^a \equiv c^b
\pmod{n}$.  \hfill \True\ \qquad \False
\examspace[0.4in]

\begin{solution}
  \False.  Need $c$ relatively prime to $n$.  Counterexample:
  $n=4$, so $\phi(n) = 2$; $a=1, b=3$, so $a \equiv b
  \pmod \phi(n)$, $c = 2$, so $c^a = 2 \not\equiv 0 = c^b \pmod 4$.
\end{solution}

\iffalse
\ppart If $a \equiv b \pmod{n}$, then $P(a) \equiv P(b) \pmod{n}$ for any
polynomial $P(x)$ with integer coefficients.
 \hfill \True\ \qquad \False\ \examspace[0.4in]
\begin{solution}
true
\end{solution}
\fi

\ppart If $a \equiv b \pmod{nm}$, then $a \equiv b \pmod{n}$, for $m,n > 1$.
 \hfill \True\ \qquad \False
\examspace[0.4in]

\begin{solution}
\True.
\end{solution}

\ppart For relatively prime $m,n >1$,\hfill \True\ \qquad \False
\[
[a \equiv b \pmod{m} \QAND a \equiv b \pmod{n}]\quad \QIFF\quad [a \equiv b \pmod{mn}]
\]
\examspace[0.4in]

\begin{solution}
\True.  The Chinese Remainder Theorem.
\end{solution}

\iffalse
\ppart Assuming $a,b$ have inverses modulo $n$, if $a^{-1} \equiv
  b^{-1} \pmod{n}$, then $a \equiv b \pmod{n}$. \hfill \True\
  \qquad \False\ \examspace[0.4in]

\begin{solution}
\True\
\end{solution}
\fi

\ppart If $a,b >1$, then \hfill \True\ \qquad \False
\begin{center}
[$a$ has a multiplicative inverse mod $b$ iff $b$ also has one mod $a$].
\end{center}
\examspace[0.4in]

\begin{solution}
\True.  $a$ has a multiplicative inverse mod $b$ iff
$a,b$ relatively prime iff $b$ has a multiplicative inverse mod $a$.
\end{solution}

\iffalse
\ppart If $\gcd(a,n)=1$, then $a^{n-1} \equiv 1 \pmod{n}$. \hfill
  \True\ \qquad \False
\examspace[0.4in]

\begin{solution}
\False.  Let $a=5$, $n =6$.
\end{solution}
\fi

\eparts

\end{problem}
