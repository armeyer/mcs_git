\begin{problem}
Using induction, prove that
\begin{equation}\label{n!sum}
n! = 1 + \sum_{k=1}^{n} (k-1) \cdot (k-1)!
\end{equation}
for positive integers $n$.
Clearly state the {\it induction hypothesis}
and prove the {\it base case} and the {\it inductive step}.

\examspace[3in]

\begin{solution}
\inductioncase{Induction Hypothesis}:
$P(n) \eqdef \text{Equation \ref{n!sum} holds for } n$.

\inductioncase{Base case}: ($n = 1$).
\begin{align}
1 + \sum_{k=1}^{1} (k-1) \cdot (k-1)! = 1 + 0 \cdot 0! = 1 = 1!
\end{align}

\inductioncase{Inductive step}: Assume that the induction hypothesis
holds for some $n > 0$. Then,
\begin{align}\label{n!sum_ind}
1 + \sum_{k=1}^{n+1} (k-1) \cdot (k-1)!
	& = 1 + \sum_{k=1}^{n} (k-1) \cdot (k-1)! + n \cdot n! \\
	& = n! + n \cdot n!  \quad \text{by induction hypothesis} \\
	& = (n+1) \cdot n! = (n+1)! \, ,
\end{align}
which implies that the hypothesis holds for $n+1$.
\end{solution}

\end{problem}
