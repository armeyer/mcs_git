\begin{problem}
  \Albert\ has a gambling problem.  He plays 35 hands of draw poker,
  30 hands of black jack, and 20 hands of stud poker per day.  He wins
  a hand of draw poker with probability 1/7, a hand of black jack with
  probability 1/6, and a hand of stud poker with probability 1/5.  Let
  $W$ be the expected number of hands that \Albert\ wins in a day.

\bparts

\ppart What is $\expect{W}$?

\exambox{1.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
The expectation is the sum of the expectations of all the individual
hands, namely
\[
35(1/7)+30(1/6)+20(1/5)= 14.
\]
\end{solution}

\ppart What would the Markov bound be on the probability that
\Albert\ will win at least 45 hands on a given day?

\exambox{0.5in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
The expected number of games won is 14, so by Markov, $\pr{W \geq 45}
\leq 14/45$.
\end{solution}

\ppart Assume the outcomes of the card games are pairwise independent.
What is $\variance{W}$?  You may answer with a numerical expression
that is not completely evaluated.

\exambox{3.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
With pairwise independence, the variance can also be calculated using
linearity.  For an individual hand the variance is $p(1-p)$ where $p$
is the probability of winning.  Therefore the variance is
\[
35(1/7)(6/7) + 30(1/6)(5/6) + 20(1/5)(4/5) = 2447/210 = 11\ \tfrac{137}{210} \approx 11.7
\]
\end{solution}

\ppart What would the Chebyshev bound be on the probability that
\Albert\ will win at least 45 hands on a given day?  You may answer
with a numerical expression that includes the constant $v =
\variance{W}$.

\exambox{2.0in}{0.5in}{0.0in}
\examspace[0.5in]

\begin{solution}
\[
\pr{W \geq 45} = \pr{W - 15 \geq 30} \leq \pr{\abs{W - 15} \geq 30} \leq \frac{v}{30^2}
   = \frac{2447}{210(30)^2} \approx 0.013.
\]

(A very slightly better bound comes from using the one-sided Chebyshev
bound from Problem~\bref{PS_Chebyshev_one_sided}.)

\end{solution}
\eparts
\end{problem}
