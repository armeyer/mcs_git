\begin{problem}
Let $T$ be a positive integer valued random variable such that
\[
\pdf_T(n)  = \frac{1}{an^2},
\]
where
\[
a \eqdef \sum_{n \in \integers^+} \frac{1}{n^2}.
\]

\bparts

\ppart
Prove that $\expect{T}$ is infinite.

\begin{solution}
\begin{align*}
\expect{T}
     & \eqdef \sum_{n \in \integers^{+}} n \pdf_T(n)\\
     & = \sum_{n \in \integers^{+}} n\frac{1}{an^2}\\
     & = \sum_{n \in \integers^{+}} \frac{1}{an}\\
     & = \frac{1}{a} \lim_{n \in \integers^+} H_n = \infty.
\end{align*}
\end{solution}

\ppart Prove that $\expect{\sqrt{T}}$ is finite.

\begin{solution}
	
\begin{eqnarray*}
\expect{\sqrt{T}} &=& \sum_{n \in \integers^{+}} \sqrt{n} \cdot \frac{1}{an^2} \\
&=& \sum_{n \in \integers^{+}} \frac{1}{an^{3/2}} \\
&<& \frac{1}{a} {\left ( 1 + \int_1^\infty \frac{1}{n^{3/2}} \right )} \\
&=& \frac{1}{a} + \frac{1}{a} \bigg|_{1}^\infty \frac{-2}{\sqrt{n}} \\
&=& \frac{1}{a} + \frac{1}{a} {\left ( 0 - \frac{-2}{\sqrt{1}} \right )} \\
&=& \frac{3}{a} < \infty
\end{eqnarray*}

\end{solution}

\begin{staffnotes}
Note that if we define $R=\sqrt{T}$, then $R$ has finite expectation,
but the variance of $R$ is infinite.
\end{staffnotes}

\eparts

\end{problem}
