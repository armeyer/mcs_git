\begin{problem}
\inhandout{The Fibonacci numbers $F(0) F(1) F(2), \dots$ are defined as follows:
\[
F(n) \eqdef \begin{cases}
  0               & \mbox{if $n = 0$},\\
  1               & \mbox{if $n = 1$},\\
  F(n-1) + F(n-2) & \mbox{if $n >1$}.
\end{cases}
\]}
\inbook{The Fibonacci numbers $F(n)$ are described in Section~\bref{sec:fib}.}

Prove by induction that for all $n \geq 0$,
\begin{equation}\label{sumkfk}
  %F(n-1) \cdot F(n+1) - F(n)^2 =  (-1)^n.
  \sum_{k=0}^n k F(k) = n F(n+2) - F(n+3) + 2
\end{equation}

\begin{solution}
\begin{proof}
  The proof is by induction on $n$ with equation~\eqref{sumkfk} as
  induction hypothesis $P(n)$.

\inductioncase{base case:} $n=0$.  $P(0)$ holds because
\[
  0 F(0) = 0 = 0 - 2 + 2 = 0 F(2) - F(3) + 2
\]

\inductioncase{inductive step:} We show that $P(n+1)$ is true, assuming that
$n \geq 0$ and $P(n)$ is true.  To do this, we consider
equation~\eqref{sumkfk} with $n$ replaced by $n+1$ and then
transform the left-hand side of this new equation into the right-hand
side. So starting with $n+1$ for $n$ in left
side of~\eqref{sumkfk}, we have
\begin{align*}
  \sum_{k=0}^{n+1} k F(k) &= (n+1)F(n+1)+ \sum_{k=0}^n k F(k)\\
  &= (n+1)F(n+1) + n F(n+2) - F(n+3) + 2 \\
                &\text{(induction hypothesis~\eqref{sumkfk})}\\
  &= (n+1)(F(n+1) + F(n+2)) - (F(n+2) + F(n+3)) + 2 \\
                &\text{(regroup terms)}\\
  &= (n+1) F(n+3) - F(n+4) + 2 \\
                & \text{(def.\ of $F(n+3)$ \& $F(n+4)$)}\\
%\lefteqn{F((n+1) - 1) \cdot F((n+1) + 1) - F(n+1)^2}\\
%        & =  F(n) \cdot F(n+2) - F(n+1)^2 
%             & \text{(simplify $(n+1)\pm 1$)}\\
%        & =  F(n) \cdot (F(n+1) + F(n))\\
%        & \qquad - F(n+1) \cdot (F(n) + F(n-1))
%                & \text{(def.\ of $F(n+2)$ \& $F(n+1)$)}\\
%        & =  F(n) \cdot F(n+1) + F(n) \cdot F(n)\\
%        & \qquad -  F(n+1) \cdot F(n) - F(n+1) \cdot F(n-1)
%                &\text{(distribute multipliers)}\\
%        & =  F(n) \cdot F(n) - F(n+1) \cdot F(n-1) 
%                & \text{(cancel $F(n) \cdot F(n+1$))}\\
%        & =  - (-1)^n
%                &\text{(induction hypothesis~\eqref{sumkfk})}\\
%        & =  (-1)^{n+1}.
\end{align*}
This proves $P(n+1)$, completing the inductive step.

We conclude by induction that equation~\eqref{sumkfk} holds for all
$n \geq 0$.
\end{proof}
\end{solution}

\end{problem}
