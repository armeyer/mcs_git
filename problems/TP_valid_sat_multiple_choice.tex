\begin{problem}
  Indicate whether each of the following propositional formulas is valid
  (V), satisfiable but not valid (S), or not satisfiable (N).  For the
  satisfiable ones, indicate a satisfying truth assignment.

\begin{align*}
M           & \QIMP  Q  \\ 
M           & \QIMP  (\bar{P} \QOR \bar{Q})  \\ 
M           & \QIMP  [M \QAND (P \QIMP M)]  \\ 
(P \QOR Q)  & \QIMP  Q  \\
(P \QOR Q)  & \QIMP  (\bar{P} \QAND \bar{Q})  \\
(P \QOR Q)  & \QIMP  [M \QAND (P \QIMP M)] \\
(P \QXOR Q) & \QIMP  Q  \\
(P \QXOR Q) & \QIMP  (\bar{P} \QOR \bar{Q})  \\
(P \QXOR Q) & \QIMP  [M \QAND (P \QIMPLIES M)]
\end{align*}

\begin{solution}
\[
\begin{array}{r l l l} 
M & \qimplies & Q & S\\ 
M & \qimplies & \bar{P} \QOR \bar{Q} & S\\ 
M & \qimplies & M \QAND (P \QIMPLIES M) & V\\ 
P \QOR Q & \qimplies & Q & S\\
P \QOR Q & \qimplies & \bar{P} \QAND \bar{Q} & N\\
P \QOR Q & \qimplies & M \QAND (P \QIMPLIES M) & S\\
P \QXOR Q & \qimplies & Q & S\\
P \QXOR Q & \qimplies & \bar{P} \QOR \bar{Q} & V\\
P \QXOR Q & \qimplies & M \QAND (P \QIMPLIES M) & S\\
\end{array} 
\]
\end{solution}

\end{problem}
