\begin{problem}
There are lots of different sizes of infinite sets.  For example, starting
with the infinite set $\nngint$ of nonnegative integers, we can build
the infinite sequence of sets
\[
\nngint\ \strict\ \power(\nngint)\ \strict\ \power(\power(\nngint))\
\strict\ \power(\power(\power(\nngint)))\ \strict \dots.
\]
where each set is ``strictly smaller'' than the next one by
Theorem~\bref{powbig}.  Let ${\power}^n(\nngint)$ be the $n$th set in
the sequence, and
  \[
  U \eqdef \lgunion_{n=0}^\infty {\power}^n(\nngint).
  \]

\bparts

\ppart Prove that
\begin{equation}\label{Usp} 
U \surj {\power}^n(\nngint),
\end{equation}
for all $n >0$.

\begin{solution}
Everything follows from a trivial observation: if $A \supseteq B$, then $A
\surj B$.  (Why is this trivial?)

Since $U \supseteq {\power}^n(\nngint)$, we now immediately
have~\eqref{Usp}.
\end{solution}

\ppart  Prove that
\[
{\power}^n(\nngint) \strict U
\]
for all $n \in \nngint$.

Now of course, we could take $U, \power(U), \power(\power(U)), \dots$
and keep on in this way building still bigger infinities indefinitely.

\begin{solution}
We must show that
\begin{equation}\label{nopsU}
\QNOT({\power}^n(\nngint) \surj U)
\end{equation}
for all $n \in \nngint$.

To prove~\eqref{nopsU}, assume to the contrary that ${\power}^m(\nngint)
\surj U$ for some $m$.  Now we know from~\eqref{Usp} that $U \surj
{\power}^{m+1}(\nngint)$.  But this implies that
\[
{\power}^m(\nngint)
\surj {\power}^{m+1}(\nngint) = \power({\power}^m(\nngint)).
\]
That is, $A \surj \power(A)$ where $A = {\power}^m(\nngint)$, which
contradicts Cantor's Theorem~\bref{powbig}.
\end{solution}

\eparts
\end{problem}
