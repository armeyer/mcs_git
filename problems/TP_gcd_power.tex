\begin{problem}
Prove that
\[
\gcd(a^5,b^5) = \gcd(a,b)^5
\]
for all $a,b \in \integers$.

\begin{solution}
\begin{proof}
Let $\expof{k}{n}$ be the largest power of $k$ that divides $n$, that is,
\[
\expof{k}{n} \eqdef \max\set{i \suchthat k^i\text{ divides } n}.
\]
Hence,
\begin{equation}
\expof{k}{mn} = \expof{k}{m} + \expof{k}{n}.\tag{*}
\end{equation}

By the Prime Factorization Theorem, we know that
\[
\expof{p}{\gcd(m,n)} = \min(\expof{p}{m},\expof{p}{n}),
\]
and so
\[
\expof{p}{a^5} = 5\expof{p}{a},
\]
by~(*).

\end{proof}
\end{solution}

\end{problem}
