\begin{problem}Prove by induction that
\begin{equation}\label{s1nfac}
\sum_1^n k\cdot k! = (n+1)! - 1.
\end{equation}
\end{problem}

\begin{solution}
The induction hypothesis $P(n)$ will be~\eqref{s1nfac}
\begin{proof}

\inductioncase{Base case} ($n=1$): both sides equal 1.

\inductioncase{inductive step.}
\begin{align*}
\sum_1^{n+1} k\cdot k!
  & = \sum_1^{n} k\cdot k! + n(n+1)!\\
  & =((n+1)! - 1) + (n+1)(n+1)!
     & \text{(induction hypothesis)}\\
  & = (n+1)!(1 + (n+1)) - 1
     & \text{factoring out }(n=1)!\\
  & = (n+1)!(n+2) - 1\\
  & + (n+2)!-1.
\end{align*}
This proves $P(n+1)$, as required.
\end{proof}
\end{solution}
