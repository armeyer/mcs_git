\begin{problem}
How many numbers in the interval $\Zintv{1}{1100}$ are divisible by 2,
5, or 11?

\hint Let $D_n$ be the numbers in the interval that are divisible by $n$.

\examspace[3.5in]

\begin{solution}
The set of numbers divisible by 2, 5, or 11 equals $D_2 \cup D_5 \cup
D_{11}$.  By the Inclusion-Exclusion Rule, the size of this set is:
\begin{align*}
\card{D_2 \cup D_5 \cup D_{11}}
    & = \card{D_2} + \card{D_5} + \card{D_{11}} \\
    & \quad - \card{D_2 \cap D_5} - \card{D_2 \cap D_{11}} - \card{D_5
      \cap D_{11}}\\
    & \quad + \card{D_2 \cap D_5 \cap D_{11}} \\
    & = \frac{1100}{2} + \frac{1100}{5} + \frac{1100}{11}
        - \frac{1100}{2 \cdot 5} - \frac{1100}{2 \cdot 11} - \frac{1100}{5
          \cdot 11}
        + \frac{1100}{2 \cdot 5 \cdot 11} \\
    & = 550 + 220 + 100 - 110 - 50 - 20 +10\\
    & = 700.
\end{align*}
\end{solution}

\end{problem}
