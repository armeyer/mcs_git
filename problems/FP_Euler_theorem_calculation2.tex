\begin{problem} \mbox{}
\bparts
\ppart What is the prime factorization of $220$? 
\begin{center}
\examrule[2.0in]
\end{center}

\begin{solution}
$202 = 2^2 \cdot 5 \cdot 11$
\end{solution}

\ppart What is $\phi(220)$, where $\phi$ is Euler's function? \hfill \examrule

\begin{solution}
\begin{align*}
\phi(220) & = \phi(2^2 \cdot 5 \cdot 11)\\
          & = \phi(2^2 \cdot 5)\phi(11)
                 & \text{(since $\gcd(2^2\cdot 5,11)=1$)}\\
          & = \phi(2^2)\phi(5)\phi(11) 
                 & \text{(since $\gcd(2^2,5)=1$)}\\
          & = (2^2-2^1)(5-1)(11-1)
                 & \text{(since $2$, $5$ and $11$ are prime)}\\
          & = (2)(4)(10) = 80.
\end{align*}

\end{solution}

\ppart What is the remainder of $63^{7201}$ divided by $220$? \hfill \examrule

\iffalse
\hint{$9601= (80 \cdot 120) + 1$; \idx{Euler's theorem}}
\fi

\begin{solution} 
\[
63.
\]

Since $63=3^2 \cdot 7$ and $220=2^2 \cdot 5 \cdot 11$ are relatively prime,
Euler's theorem implies that
\[
63^{\phi(220)} = 1 \inzmod{220}.
\]

Therefore $63^{80}\equiv 1 \pmod{220}$.

Since $7201 = 80 \cdot 900 + 1$,
%
\begin{align*}
63^{9601} & = 63 \cdot 63^{80 \cdot 90}\\
             & = 63 \cdot 1^{90}\\
             & = 63 \inzmod{220}.
\end{align*}
So the remainder of $63^{7201}$ divided by $220$ is $63$.
\end{solution}

\eparts

\end{problem}
