%% title: Set Operations

Let $A = \set{a,b,c,d,e}$ and $B = \set{a,b,c,d,e,f,g,h}$.  Find:

\bparts

\ppart
$A \union B$

\begin{solution}
$\set{ a, b, c, d, e, f, g, h }$
\end{solution}

\ppart
$A \intersect B$

\begin{solution}
$\set{ a, b, c, d, e }$
\end{solution}

\ppart
$A - B$

\begin{solution}
$\emptyset$
\end{solution}

\ppart
$B - A$

\begin{solution}
$\set{ f, g, h }$
\end{solution}
