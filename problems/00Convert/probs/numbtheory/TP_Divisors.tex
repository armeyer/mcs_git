\bparts

\ppart
How many prime divisors does 12 have?

\begin{solution}
Two: 2 and 3.
\end{solution}

\ppart
How many positive divisors does 12 have?

\begin{solution}
Six: 1, 2, 3, 4, 6, and 12.
\end{solution}

\ppart
How many divisors does 12 have?

\begin{solution}
Twelve: $\pm 1$,  $\pm 2$,  $\pm 3$,  $\pm 4$,  $\pm 6$, and  $\pm 12$.
\end{solution}

\eparts

\end{problem}
