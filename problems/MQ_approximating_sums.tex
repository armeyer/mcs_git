\begin{problem}

%\textbf{I WORRY...} that bounds that do not hold in general may actually magically
%work in some of the cases below where I say they won't.

\begin{problemparts}

\problempart
Let
\[
R \eqdef \sum_{i=1}^n{\log_2{(i+1)}}.
\]
Which of the following are suitable bounds for $R$?
\begin{enumerate}
\item $\displaystyle R\leq\int_2^{n+2} \! \log_2{x} \, \mathrm{d}x$  % Good
\item $\displaystyle R\leq \log_2{n}+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item $\displaystyle R\leq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item $\displaystyle R\geq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Good
\end{enumerate}

\begin{solution}
\begin{enumerate}
\item GOOD: $\displaystyle R\leq\int_2^{n+2} \! \log_2{x} \, \mathrm{d}x$  % Good
\item BAD: $\displaystyle R\leq \log_2{n}+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item BAD: $\displaystyle R\leq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item GOOD: $\displaystyle R\geq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Good
\end{enumerate}

Let $f(x) = \log_2{(x+1)}$.  $f$ is a strictly increasing function for all $x \geq 0$.
Therefore, 
\begin{align*}
  R &= \sum_{i=1}^n{f(i)} \\
    &\leq \int_0^n \! f(x+1) \, \mathrm{d}x \\
    &= \int_0^n \! \log_2{(x+2)} \, \mathrm{d}x \\
    &= \int_2^{n+2} \! \log_2{x} \, \mathrm{d}x
\end{align*}
Also,
\begin{align*}
  R &= \sum_{i=1}^n{f(i)} \\
    &\geq f(1) + \int_1^n \! f(x) \, \mathrm{d}x \\
    &= \log_2{(1+1)} + \int_1^n \! \log_2{(x+1)} \, \mathrm{d}x \\
    &= 1 + \int_1^n \! \log_2{(x+1)} \, \mathrm{d}x \\
\end{align*}
\end{solution}

\problempart
Let
\[
S \eqdef \sum_{i=1}^n i.
\]
Which of the following are suitable bounds for $S$?
\begin{enumerate}
\item $\displaystyle S\leq\int_1^{n+1} \! \ceil{x} \, \mathrm{d}x$  % Good
\item $\displaystyle S\leq n +\int_0^{n-1} \! \ceil{x+1} \, \mathrm{d}x$  % Good
\item $\displaystyle S\leq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x$  % Good
\item $\displaystyle S\geq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x$  % Good
\end{enumerate}

\begin{solution}
\begin{enumerate}
\item GOOD: $\displaystyle S\leq\int_1^{n+1} \! \ceil{x} \, \mathrm{d}x$  % Good
\item GOOD: $\displaystyle S\leq n +\int_0^{n-1} \! \ceil{x+1} \, \mathrm{d}x$  % Good
\item GOOD: $\displaystyle S\leq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x$  % Good
\item GOOD: $\displaystyle S\geq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x$  % Good
\end{enumerate}

Clearly,
\[
S \leq \int_1^{n+1} \! \ceil{x} \, \mathrm{d}x
\]
and
\[
S \leq n +\int_0^{n-1} \! \ceil{x+1} \, \mathrm{d}x
\]
and
\[
S\geq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x
\]
hold because $\ceil{x}$ is a weakly increasing function of $x$ for all
$x\geq 0$.  (Note that $\ceil{1} = 1$ and $\ceil{n} = n$.)

\[
S \leq 1+\int_1^n \! \ceil{x} \, \mathrm{d}x
\]
holds as well because the staircase form of the graph of the ceiling
function results in the equality
\[
S = 1+\int_1^n \! \lceil{x}\rceil \, \mathrm{d}x
\]
\end{solution}

\problempart
Let
\[
T \eqdef \sum_{i=1}^n{\cos{\frac{\pi i}{100}}},
\]
where $2\leq n \leq 80$.  Which of the following are suitable bounds
for $T$?
\begin{enumerate}
\item $\displaystyle T\leq\int_0^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\leq  \cos{\frac{\pi}{100}} + \cos{\frac{\pi}{50}} + \int_2^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\geq \int_1^{n+1} \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\geq \cos{\frac{\pi}{100}} + \int_1^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Bad
\end{enumerate}

\begin{solution}
\begin{enumerate}
\item GOOD: $\displaystyle T\leq\int_0^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item GOOD: $\displaystyle T\leq  \cos{\frac{\pi}{100}} + \cos{\frac{\pi}{50}} + \int_2^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item GOOD: $\displaystyle T\geq \int_1^{n+1} \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item BAD: $\displaystyle T\geq \cos{\frac{\pi}{100}} + \int_1^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Bad
\end{enumerate}

The correct bounds follow from the fact that $\cos{\frac{\pi x}{100}}$
is a strictly decreasing function for $2 \leq x < 81$.
\end{solution}

\problempart How would your answer to the preceding part change if the
only restriction on $n$ was $n\geq 2$?  In a single short sentence,
explain why.

\begin{solution}
$\cos{\frac{\pi x}{100}}$ is not weakly decreasing for all $x \geq 2$,
  nor is it weakly increasing for all $x \geq 2$.  Thus none of the
  bounds given in the previous part would hold on this interval.
\end{solution}

\end{problemparts}
\end{problem}
